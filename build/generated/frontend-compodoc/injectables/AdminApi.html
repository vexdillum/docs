import {inject, Injectable} from '@angular/core';
import {HttpClient, HttpParams} from '@angular/common/http';
import {ApiConfig} from '../../shared/config/api.config';
import {
  AdminAuditLogListResponse,
  AdminChallengeListResponse,
  AdminChallengeResponse,
  AdminConfirmRequest,
  AdminContestResponse,
  AdminContestUpdateRequest,
  AdminCreateChallengeRequest,
  AdminDeleteUserResponse,
  AdminInstanceListResponse,
  AdminInstanceResponse,
  AdminInstanceStatusRequest,
  AdminInstanceTTLRequest,
  AdminOverviewResponse,
  AdminSolveListResponse,
  AdminTeamListResponse,
  AdminUpdateChallengeRequest,
  AdminUserListResponse,
  AdminUserResponse,
  ChallengeAccessResponse,
  ChallengeLogsResponse,
  ChallengeStatusResponse,
  DeleteResponse,
  LeaderboardFreezeResponse,
} from '../../entities/admin/admin.interface';

@Injectable({
  providedIn: 'root',
})
export class AdminApi {
  private readonly http = inject(HttpClient);
  private readonly baseApiUrl = inject(ApiConfig).getBaseApiUrl() + 'admin/';

  // Overview
  getOverview() {
    return this.http.get<AdminOverviewResponse>(`${this.baseApiUrl}overview`, {
      withCredentials: true,
    });
  }

  // Audit
  getAuditLogs() {
    return this.http.get<AdminAuditLogListResponse>(`${this.baseApiUrl}audit`, {
      withCredentials: true,
    });
  }

  // Challenges
  getChallenges() {
    return this.http.get<AdminChallengeListResponse>(`${this.baseApiUrl}challenges`, {
      withCredentials: true,
    });
  }

  getChallenge(challengeId: string) {
    return this.http.get<AdminChallengeResponse>(`${this.baseApiUrl}challenges/${challengeId}`, {
      withCredentials: true,
    });
  }

  createChallenge(payload: AdminCreateChallengeRequest) {
    return this.http.post<AdminChallengeResponse>(`${this.baseApiUrl}challenges`, payload, {
      withCredentials: true,
    });
  }

  updateChallenge(challengeId: string, payload: AdminUpdateChallengeRequest) {
    return this.http.put<AdminChallengeResponse>(`${this.baseApiUrl}challenges/${challengeId}`, payload, {
      withCredentials: true,
    });
  }

  deleteChallenge(challengeId: string, payload: AdminConfirmRequest) {
    return this.http.request<DeleteResponse>('delete', `${this.baseApiUrl}challenges/${challengeId}`, {
      body: payload,
      withCredentials: true,
    });
  }

  getChallengeAccess(challengeId: string) {
    return this.http.get<ChallengeAccessResponse>(`${this.baseApiUrl}challenges/${challengeId}/access`, {
      withCredentials: true,
    });
  }

  deployChallenge(challengeId: string) {
    return this.http.post<AdminChallengeResponse>(`${this.baseApiUrl}challenges/${challengeId}/deploy`, {}, {
      withCredentials: true,
    });
  }

  getChallengeLogs(challengeId: string) {
    return this.http.get<ChallengeLogsResponse>(`${this.baseApiUrl}challenges/${challengeId}/logs`, {
      withCredentials: true,
    });
  }

  getChallengeStatus(challengeId: string) {
    return this.http.get<ChallengeStatusResponse>(`${this.baseApiUrl}challenges/${challengeId}/status`, {
      withCredentials: true,
    });
  }

  stopChallenge(challengeId: string) {
    return this.http.post<AdminChallengeResponse>(`${this.baseApiUrl}challenges/${challengeId}/stop`, {}, {
      withCredentials: true,
    });
  }

  // Contest
  getContest() {
    return this.http.get<AdminContestResponse>(`${this.baseApiUrl}contest`, {
      withCredentials: true,
    });
  }

  updateContest(payload: AdminContestUpdateRequest) {
    return this.http.put<AdminContestResponse>(`${this.baseApiUrl}contest`, payload, {
      withCredentials: true,
    });
  }

  startContest() {
    return this.http.post<AdminContestResponse>(`${this.baseApiUrl}contest/start`, {}, {
      withCredentials: true,
    });
  }

  endContest(payload: AdminConfirmRequest) {
    return this.http.post<AdminContestResponse>(`${this.baseApiUrl}contest/end`, payload, {
      withCredentials: true,
    });
  }

  freezeContest() {
    return this.http.post<AdminContestResponse>(`${this.baseApiUrl}contest/freeze`, {}, {
      withCredentials: true,
    });
  }

  unfreezeContest() {
    return this.http.post<AdminContestResponse>(`${this.baseApiUrl}contest/unfreeze`, {}, {
      withCredentials: true,
    });
  }

  // Instances
  getInstances(limit?: number) {
    const options = {
      params: limit ? new HttpParams().set('limit', limit) : undefined,
      withCredentials: true,
    };

    return this.http.get<AdminInstanceListResponse>(`${this.baseApiUrl}instances`, options);
  }

  getInstance(instanceId: string) {
    return this.http.get<AdminInstanceResponse>(`${this.baseApiUrl}instances/${instanceId}`, {
      withCredentials: true,
    });
  }

  deleteInstance(instanceId: string, payload: AdminConfirmRequest) {
    return this.http.request<DeleteResponse>('delete', `${this.baseApiUrl}instances/${instanceId}`, {
      body: payload,
      withCredentials: true,
    });
  }

  extendInstance(instanceId: string, payload: AdminInstanceTTLRequest) {
    return this.http.post<AdminInstanceResponse>(`${this.baseApiUrl}instances/${instanceId}/extend`, payload, {
      withCredentials: true,
    });
  }

  restartInstance(instanceId: string, payload: AdminInstanceTTLRequest) {
    return this.http.post<AdminInstanceResponse>(`${this.baseApiUrl}instances/${instanceId}/restart`, payload, {
      withCredentials: true,
    });
  }

  updateInstanceStatus(instanceId: string, payload: AdminInstanceStatusRequest) {
    return this.http.post<AdminInstanceResponse>(`${this.baseApiUrl}instances/${instanceId}/status`, payload, {
      withCredentials: true,
    });
  }

  // Leaderboard freeze
  getLeaderboardFreeze() {
    return this.http.get<LeaderboardFreezeResponse>(`${this.baseApiUrl}leaderboard/freeze`, {
      withCredentials: true,
    });
  }

  freezeLeaderboard() {
    return this.http.post<LeaderboardFreezeResponse>(`${this.baseApiUrl}leaderboard/freeze`, {}, {
      withCredentials: true,
    });
  }

  unfreezeLeaderboard() {
    return this.http.delete<LeaderboardFreezeResponse>(`${this.baseApiUrl}leaderboard/freeze`, {
      withCredentials: true,
    });
  }

  // Solves
  getSolves() {
    return this.http.get<AdminSolveListResponse>(`${this.baseApiUrl}solves`, {
      withCredentials: true,
    });
  }

  // Teams
  getAdminTeams() {
    return this.http.get<AdminTeamListResponse>(`${this.baseApiUrl}teams`, {
      withCredentials: true,
    });
  }

  // Users
  getAdminUsers() {
    return this.http.get<AdminUserListResponse>(`${this.baseApiUrl}users`, {
      withCredentials: true,
    });
  }

  getAdminUser(userId: string) {
    return this.http.get<AdminUserResponse>(`${this.baseApiUrl}users/${userId}`, {
      withCredentials: true,
    });
  }

  deleteAdminUser(userId: string, payload: AdminConfirmRequest) {
    return this.http.request<AdminDeleteUserResponse>('delete', `${this.baseApiUrl}users/${userId}`, {
      body: payload,
      withCredentials: true,
    });
  }

  banUser(userId: string) {
    return this.http.post<AdminUserResponse>(`${this.baseApiUrl}users/${userId}/ban`, {}, {
      withCredentials: true,
    });
  }

  unbanUser(userId: string) {
    return this.http.post<AdminUserResponse>(`${this.baseApiUrl}users/${userId}/unban`, {}, {
      withCredentials: true,
    });
  }
}

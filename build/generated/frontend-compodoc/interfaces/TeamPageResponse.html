import {
  AfterViewInit,
  Component,
  inject,
  Input,
  OnChanges,
  signal,
  SimpleChanges,
  TemplateRef,
  ViewChild
} from '@angular/core';
import {Location, DatePipe} from '@angular/common';
import {ReactiveFormsModule} from '@angular/forms';
import {RouterLink} from '@angular/router';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  ContextMenuModule,
  InputModule,
  NotificationModule,
  TagModule,
  TableModel,
  TableModule,
  TilesModule
} from 'carbon-components-angular';
import {catchError, finalize, forkJoin, Observable, of, switchMap} from 'rxjs';
import {TeamInterface, TeamResponseInterface, TeamSolveInterface} from '../../entities/team.interface';
import {TeamApi} from '../../utils/team/team-api';
import {UserInterface} from '../../entities/user.interface';
import {MeApi} from '../../utils/auth/me-api';
import {EditTeamModal} from '../../widgets/edit-team-modal/edit-team-modal';
import {
  createMembersModel,
  createSolvesModel,
  getCaptainBadge,
  MemberBadge,
  MemberTableItem
} from '../../utils/team/team-table-models';

interface TeamPageData {
  team: TeamInterface | null;
  solves: TeamSolveInterface[];
  members: UserInterface[];
  me: UserInterface | null;
}

interface TeamPageResponse {
  team: TeamResponseInterface | null;
  solves: { solves: TeamSolveInterface[] } | null;
  members: { members: UserInterface[] } | null;
  me: { user: UserInterface, team?: TeamInterface | null } | null;
}

@Component({
  selector: 'app-team-page',
  imports: [
    ButtonModule,
    ContextMenuModule,
    InputModule,
    NotificationModule,
    ReactiveFormsModule,
    RouterLink,
    TagModule,
    TranslocoDirective,
    TilesModule,
    TableModule,
    EditTeamModal
  ],
  templateUrl: './team-page.html',
  styleUrl: './team-page.scss',
  providers: [DatePipe],
})
export class TeamPage implements AfterViewInit, OnChanges {
  @Input() team_id?: string;
  @ViewChild('memberCellTemplate')
  protected memberCellTemplate!: TemplateRef<{ data: MemberTableItem }>;

  private readonly teamApi = inject(TeamApi);
  private readonly meApi = inject(MeApi);
  private readonly datePipe = inject(DatePipe);
  private readonly location = inject(Location);
  private readonly transloco = inject(TranslocoService);

  readonly team = signal<TeamInterface | null>(null);
  readonly solvesList = signal<TeamSolveInterface[] | null>(null);
  readonly membersList = signal<UserInterface[] | null>(null);
  readonly myUser = signal<UserInterface | null>(null);
  readonly loading = signal<boolean>(true);
  readonly deletingMemberId = signal<string | null>(null);
  readonly isMyTeam = signal<boolean>(false);
  readonly isCaptain = signal<boolean>(false);
  readonly editTeamModalOpen = signal<boolean>(false);

  solves = new TableModel();
  members = new TableModel();

  ngAfterViewInit(): void {
    this.loadPage();
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes['team_id'] && !changes['team_id'].firstChange) {
      this.loadPage();
    }
  }

  private loadPage() {
    const teamId = this.team_id ?? null;

    this.resetPageState();
    this.fetchTeamPageData(teamId)
      .pipe(finalize(() => this.loading.set(false)))
      .subscribe((res) => this.applyPageData(res));
  }

  private resetPageState(): void {
    this.loading.set(true);
    this.team.set(null);
    this.solvesList.set(null);
    this.membersList.set(null);
    this.myUser.set(null);
    this.deletingMemberId.set(null);
    this.isMyTeam.set(false);
    this.isCaptain.set(false);
    this.editTeamModalOpen.set(false);
    this.solves = this.buildSolvesModel([]);
    this.members = this.buildMembersModel([]);
  }

  private fetchTeamPageData(teamId: string | null): Observable<TeamPageResponse> {
    return forkJoin({
      team: this.teamApi.getTeam(teamId).pipe(
        catchError(() => {
          this.location.back();
          return of(null);
        })
      ),
      me: this.meApi.me().pipe(catchError(() => of(null))),
    })
      .pipe(
        switchMap(({team: teamResponse, me}) => {
          if (!teamResponse?.team) {
            return of({
              team: null,
              solves: null,
              members: null,
              me,
            });
          }

          const activeTeam = teamResponse.team;
          const isCurrentTeam = !teamId || me?.team?.id === activeTeam.id;

          return forkJoin({
            team: of(teamResponse),
            solves: this.teamApi.getTeamSolves(isCurrentTeam ? null : activeTeam.id).pipe(
              catchError(() => of(null))
            ),
            members: this.teamApi.getTeamMembers(isCurrentTeam ? null : activeTeam.id).pipe(
              catchError(() => of(null))
            ),
            me: of(me)
          });
        })
      );
  }

  private applyPageData(res: TeamPageResponse): void {
    const pageData = this.normalizePageData(res);

    this.team.set(pageData.team);
    this.membersList.set(pageData.members);
    this.myUser.set(pageData.me);
    this.solvesList.set(pageData.solves);
    this.updateOwnershipState(pageData.members, pageData.me);
    this.rebuildTableModels();
  }

  private normalizePageData(res: TeamPageResponse): TeamPageData {
    return {
      team: res.team?.team ?? null,
      solves: res.solves?.solves ?? [],
      members: res.members?.members ?? [],
      me: res.me?.user ?? null,
    };
  }

  private updateOwnershipState(members: UserInterface[], me: UserInterface | null): void {
    this.isMyTeam.set(members.some((member) => member.id === me?.id));
    this.isCaptain.set(members.some((member) => (
      member.id === me?.id && member.is_captain
    )));
  }

  openEditTeamModal() {
    if (!this.isCaptain()) {
      return;
    }

    this.editTeamModalOpen.set(true);
  }

  closeEditTeamModal() {
    this.editTeamModalOpen.set(false);
  }

  onTeamUpdated(team: TeamInterface) {
    this.team.set(team);
    this.loadPage();
  }

  onTeamDeleted() {
    this.loadPage();
  }

  removeMember(member: UserInterface) {
    if (!this.canDeleteMember(member) || this.deletingMemberId()) {
      return;
    }

    this.deletingMemberId.set(member.id);
    this.rebuildTableModels();

    this.teamApi.deleteTeamMember(member.id)
      .pipe(finalize(() => this.deletingMemberId.set(null)))
      .subscribe({
        next: () => this.loadPage(),
        error: () => this.rebuildTableModels(),
      });
  }

  private rebuildTableModels() {
    this.solves = this.buildSolvesModel(this.solvesList() ?? []);
    this.members = this.buildMembersModel(this.membersList() ?? []);
  }

  private buildSolvesModel(solves: TeamSolveInterface[]) {
    return createSolvesModel(solves, this.transloco, this.datePipe, true);
  }

  private buildMembersModel(members: UserInterface[]) {
    return createMembersModel(
      members,
      this.transloco,
      this.memberCellTemplate,
      (member) => this.getMemberBadges(member),
      {
        canDelete: (member) => this.canDeleteMember(member),
        deletingMemberId: this.deletingMemberId(),
        onDelete: (member) => this.removeMember(member),
      }
    );
  }

  private getMemberBadges(member: UserInterface): MemberBadge[] {
    return getCaptainBadge(member, this.transloco);
  }

  private canDeleteMember(member: UserInterface): boolean {
    return this.isCaptain() && member.id !== this.myUser()?.id && !member.is_captain;
  }
}

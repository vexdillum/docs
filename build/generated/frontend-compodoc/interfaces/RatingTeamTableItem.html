import {AfterViewInit, Component, computed, inject, signal, TemplateRef, ViewChild} from '@angular/core';
import {RouterLink} from '@angular/router';
import {Location} from '@angular/common';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {TableHeaderItem, TableItem, TableModel, TableModule} from 'carbon-components-angular';
import {catchError, finalize, of} from 'rxjs';
import {ScoreboardChart, ScoreboardChartTeam} from '../../widgets/charts/scoreboard-chart/scoreboard-chart';
import {TeamApi} from '../../utils/team/team-api';
import {LeaderboardSolveInterface, LeaderboardTeamInterface} from '../../entities/team.interface';

interface RatingTeamTableItem {
  name: string;
  link: string[];
}

@Component({
  selector: 'app-rating-page',
  imports: [
    RouterLink,
    TranslocoDirective,
    TableModule,
    ScoreboardChart
  ],
  templateUrl: './rating-page.html',
  styleUrl: './rating-page.scss',
})
export class RatingPage implements AfterViewInit {
  @ViewChild('teamCellTemplate')
  protected teamCellTemplate!: TemplateRef<{ data: RatingTeamTableItem }>;

  private readonly teamApi = inject(TeamApi);
  private readonly location = inject(Location);
  private readonly transloco = inject(TranslocoService);

  readonly loading = signal<boolean>(true);
  readonly chartTeams = signal<ScoreboardChartTeam[]>([]);
  readonly frozen = signal<boolean>(false);
  readonly frozenAt = signal<string | null>(null);

  ratingTitle = computed(() => {
    if (!this.frozen()) {
      return this.transloco.translate('rating-page.title');
    }
    const date = this.frozenAt() ? new Date(this.frozenAt()!) : null;
    const formatted = date
      ? date.toLocaleString(this.transloco.getActiveLang(), {
          hour: '2-digit',
          minute: '2-digit',
          day: 'numeric',
          month: 'long',
        })
      : '';
    return this.transloco.translate('rating-page.frozenTitle', { time: formatted });
  });

  rating = this.createRatingModel([]);

  ngAfterViewInit(): void {
    this.loadData();
  }

  private loadData(): void {
    this.loading.set(true);
    this.chartTeams.set([]);
    this.rating = this.createRatingModel([]);

    this.teamApi.getLeaderboard()
      .pipe(
        catchError(() => {
          this.location.back();
          return of(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((leaderboard) => {
        this.frozen.set(leaderboard?.frozen ?? false);
        this.frozenAt.set(leaderboard?.frozen_at ?? null);

        const teams = this.sortTeams(leaderboard?.teams ?? []);

        this.rating = this.createRatingModel(teams);
        this.chartTeams.set(this.createChartTeams(teams, leaderboard?.last_solves ?? []));
      });
  }

  private sortTeams(teams: LeaderboardTeamInterface[]): LeaderboardTeamInterface[] {
    return [...teams].sort((a, b) => {
      if (a.rank && b.rank) {
        return a.rank - b.rank;
      }

      return (b.score ?? 0) - (a.score ?? 0);
    });
  }

  private createChartTeams(
    teams: LeaderboardTeamInterface[],
    solves: LeaderboardSolveInterface[]
  ): ScoreboardChartTeam[] {
    const solvesByTeam = new Map<string, LeaderboardSolveInterface[]>();

    solves.forEach((solve) => {
      const teamSolves = solvesByTeam.get(solve.team_id) ?? [];

      teamSolves.push(solve);
      solvesByTeam.set(solve.team_id, teamSolves);
    });

    return teams.slice(0, 10).map((team) => ({
      name: team.team_name,
      solves: (solvesByTeam.get(team.team_id) ?? []).map((solve) => ({
        value: solve.points,
        date: solve.solved_at,
      })),
    }));
  }

  private createRatingModel(teams: LeaderboardTeamInterface[]) {
    const model = new TableModel();

    model.header = [
      new TableHeaderItem({data: this.transloco.translate('rating-page.columns.place')}),
      new TableHeaderItem({data: this.transloco.translate('rating-page.columns.team')}),
      new TableHeaderItem({data: this.transloco.translate('rating-page.columns.solves')}),
      new TableHeaderItem({data: this.transloco.translate('rating-page.columns.score')})
    ];

    model.data = teams.map((team, index) => [
      new TableItem({data: team.rank ?? index + 1}),
      new TableItem({
        data: {
          name: team.team_name,
          link: ['/team', team.team_id],
        },
        template: this.teamCellTemplate
      }),
      new TableItem({data: team.solves_count}),
      new TableItem({data: team.score}),
    ]);

    return model;
  }
}

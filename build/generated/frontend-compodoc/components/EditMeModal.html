import {Component, EventEmitter, Input, Output, inject, signal} from '@angular/core';
import {HttpErrorResponse} from '@angular/common/http';
import {FormControl, FormGroup, ReactiveFormsModule} from '@angular/forms';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  InputModule,
  ModalModule,
  NotificationModule
} from 'carbon-components-angular';
import {catchError, finalize, of} from 'rxjs';
import {UserInterface} from '../../entities/user.interface';
import {MeInterface, UpdateMeInterface} from '../../entities/auth/me.interface';
import {MeApi} from '../../utils/auth/me-api';
import {controlInvalid, getControlErrorKey, loginValidators, registerValidators} from '../../utils/auth/auth-form-validators';

type EditMeControlName = 'email' | 'username';

@Component({
  selector: 'app-edit-me-modal',
  imports: [
    ButtonModule,
    InputModule,
    ModalModule,
    NotificationModule,
    ReactiveFormsModule,
    TranslocoDirective
  ],
  templateUrl: './edit-me-modal.html',
  styleUrl: './edit-me-modal.scss',
})
export class EditMeModal {
  private readonly meApi = inject(MeApi);
  private readonly transloco = inject(TranslocoService);

  readonly isOpen = signal<boolean>(false);
  readonly user = signal<UserInterface | null>(null);
  readonly loading = signal<boolean>(false);
  readonly notification = signal<{ type: 'error' | 'success'; title: string; message: string } | null>(null);

  readonly form = new FormGroup({
    email: new FormControl('', loginValidators.email),
    username: new FormControl('', registerValidators.username),
  });

  @Input() set open(value: boolean) {
    this.isOpen.set(value);

    if (value) {
      this.notification.set(null);
      this.patchForm();
    }
  }

  @Input() set activeUser(value: UserInterface | null) {
    this.user.set(value);

    if (this.isOpen()) {
      this.patchForm();
    }
  }

  @Output() isCloseModal = new EventEmitter<void>();
  @Output() meUpdated = new EventEmitter<MeInterface>();

  closeModal() {
    this.isCloseModal.emit();
  }

  isInvalid(controlName: EditMeControlName) {
    return controlInvalid(this.form.get(controlName));
  }

  getErrorMessage(controlName: EditMeControlName) {
    const errorKey = getControlErrorKey(this.form.get(controlName));

    return errorKey ? this.transloco.translate(`auth-pages.validation.${controlName}.${errorKey}`) : '';
  }

  saveMe() {
    if (this.form.invalid || this.loading()) {
      this.form.markAllAsTouched();
      return;
    }

    const activeUser = this.user();

    if (!activeUser) {
      return;
    }

    const payload = this.getPayload(activeUser);

    if (!payload) {
      this.closeModal();
      return;
    }

    this.loading.set(true);
    this.notification.set(null);

    this.meApi.updateMe(payload)
      .pipe(
        catchError((error: HttpErrorResponse) => {
          this.showRequestError(error.status);
          return of(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => {
        if (!res?.user) {
          return;
        }

        this.meUpdated.emit(res);
        this.closeModal();
      });
  }

  private patchForm() {
    const activeUser = this.user();

    this.form.reset({
      email: activeUser?.email ?? '',
      username: activeUser?.username ?? '',
    });
  }

  private getPayload(activeUser: UserInterface): UpdateMeInterface | null {
    const formValue = this.form.getRawValue();
    const email = formValue.email?.trim() ?? '';
    const username = formValue.username?.trim() ?? '';

    if (email === activeUser.email && username === activeUser.username) {
      return null;
    }

    return {
      email,
      username,
    };
  }

  private showRequestError(status: number) {
    const messageKey = status === 0
      ? 'no-connection'
      : status === 400 || status === 401 || status === 403 || status === 404 || status === 409
        ? 'update-invalid'
        : 'universal';

    this.notification.set({
      type: 'error',
      title: this.transloco.translate('edit-me-modal.notifications.error.title'),
      message: this.transloco.translate(`edit-me-modal.notifications.error.${messageKey}`),
    });
  }
}

        </button>
      </cds-modal-footer>
    </cds-modal>
  }

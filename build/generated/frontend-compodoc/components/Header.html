import { CommonModule } from '@angular/common';
import { afterNextRender, Component, OnInit, computed, inject, signal } from '@angular/core';
import { LangDefinition, TranslocoDirective, TranslocoService } from '@ngneat/transloco';
import {
  ButtonModule,
  DialogModule,
  DropdownModule,
  IconModule,
  IconService,
  PlaceholderModule,
  TabsModule,
  UIShellModule
} from 'carbon-components-angular';
import { AuthApi } from '../../utils/auth/auth-api';
import { NavigationEnd, Router, RouterLink } from '@angular/router';
import { catchError, EMPTY, filter, tap } from 'rxjs';
import { MeApi } from '../../utils/auth/me-api'
import { UserInterface } from '../../entities/user.interface'

import Language16 from '@carbon/icons/es/language/16'
import Logout16 from '@carbon/icons/es/logout/16'
import User16 from '@carbon/icons/es/user/16'
import Events16 from '@carbon/icons/es/events/16'
import Settings16 from '@carbon/icons/es/settings/16'

@Component({
  selector: 'app-header',
  imports: [
    CommonModule,
    UIShellModule,
    TranslocoDirective,
    IconModule,
    DialogModule,
    PlaceholderModule,
    DropdownModule,
    ButtonModule,
    TabsModule,
    RouterLink
  ],
  templateUrl: './header.html',
  styleUrl: './header.scss',
})
export class Header implements OnInit {
  transloco = inject(TranslocoService);
  langs = this.transloco.getAvailableLangs();
  authApi = inject(AuthApi);
  meApi = inject(MeApi);
  router = inject(Router);
  private readonly iconService = inject(IconService);

  currentLang = signal<string>(localStorage.getItem('language') ?? 'en');
  isLoggedIn = signal<boolean>(false);
  meUser = signal<UserInterface | null>(null);
  isAdmin = computed(() => this.meUser()?.role === 'admin');
  sidenavExpanded = signal<boolean>(false);

  ngOnInit(): void {
    this.iconService.registerAll([
      Language16 as object,
      Logout16 as object,
      User16 as object,
      Events16 as object,
      Settings16 as object
    ]);
  }

  constructor() {
    this.transloco.setActiveLang(
      localStorage.getItem('language') ?? 'en'
    );

    afterNextRender(() => {
      this.meApi.me().pipe(
        tap((res) => {
          this.isLoggedIn.set(true);
          this.meUser.set(res.user);
        }),
        catchError(() => {
          this.isLoggedIn.set(false);
          this.meUser.set(null);
          return EMPTY;
        })
      ).subscribe();
    });

    this.router.events.pipe(
      filter(event => event instanceof NavigationEnd)
    ).subscribe(() => {
      this.sidenavExpanded.set(false);
    });
  }

  LangSelected(lang: string | LangDefinition) {
    const langId = typeof lang !== 'string' ? lang.id : lang;

    console.log(langId);

    return langId === (localStorage.getItem('language') ?? 'en');
  }

  switchLang(lang: string | LangDefinition) {
    const langId = typeof lang !== 'string' ? lang.id : lang;

    this.currentLang.set(langId);
    localStorage.setItem('language', langId);
    this.transloco.setActiveLang(langId);
  }

  logout() {
    this.isLoggedIn.set(false);
    this.meUser.set(null);
    this.authApi.logout().subscribe();
    this.sidenavExpanded.set(false);
  }

  toggleSidenav() {
    this.sidenavExpanded.update(v => !v);
  }
}

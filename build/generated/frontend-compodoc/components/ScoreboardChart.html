import {Component, Input, OnChanges, SimpleChanges, signal} from '@angular/core';
import {NgxEchartsDirective} from 'ngx-echarts';
import type { EChartsCoreOption } from 'echarts/core';

export interface ScoreboardChartSolve {
  value: number;
  date: string;
}

export interface ScoreboardChartTeam {
  name: string;
  solves: ScoreboardChartSolve[];
}

@Component({
  selector: 'app-scoreboard',
  imports: [
    NgxEchartsDirective
  ],
  templateUrl: './scoreboard-chart.html',
  styleUrl: './scoreboard-chart.scss',
})
export class ScoreboardChart implements OnChanges {
  @Input() teams: ScoreboardChartTeam[] = [];

  chartOptions = signal<EChartsCoreOption>({});

  private readonly chartTheme = {
    background: 'transparent',
    text: '#f4f4f4',
    textSecondary: '#c6c6c6',
    axis: '#8d8d8d',
    grid: '#393939',
    tooltipBackground: '#262626',
    tooltipBorder: '#525252',
    colors: [
      '#78a9ff',
      '#42be65',
      '#ffb784',
      '#ff7eb6',
      '#be95ff',
      '#3ddbd9',
      '#ee5396',
      '#82cfff',
      '#f1c21b',
      '#08bdba',
    ],
  };

  ngOnChanges(changes: SimpleChanges): void {
    if (changes['teams']) {
      this.updateChart();
    }
  }

  private updateChart(): void {
    const endTime = this.getLastSolveTime();

    this.chartOptions.set({
      ...this.createBaseOptions(endTime),
      series: this.createSeries(endTime),
      animationDuration: 350,
    });
  }

  private createBaseOptions(endTime: string | undefined): EChartsCoreOption {
    return {
      ...this.createThemeOptions(),
      grid: {
        top: 48,
        right: 32,
        bottom: 64,
        left: 56,
      },
      legend: {
        type: 'scroll',
        top: 0,
        textStyle: {
          color: this.chartTheme.textSecondary,
        },
      },
      tooltip: this.createTooltipOptions(),
      xAxis: this.createTimeAxisOptions(endTime),
      yAxis: this.createValueAxisOptions(),
      dataZoom: [
        {
          type: 'inside',
          zoomOnMouseWheel: false,
          moveOnMouseWheel: false,
        },
        {
          type: 'slider',
          height: 24,
          bottom: 16,
        }
      ],
    };
  }

  private createThemeOptions(): EChartsCoreOption {
    return {
      backgroundColor: this.chartTheme.background,
      color: this.chartTheme.colors,
      textStyle: {
        color: this.chartTheme.text,
      },
    };
  }

  private createTooltipOptions(): EChartsCoreOption['tooltip'] {
    return {
      trigger: 'axis',
      backgroundColor: this.chartTheme.tooltipBackground,
      borderColor: this.chartTheme.tooltipBorder,
      textStyle: {
        color: this.chartTheme.text,
      },
    };
  }

  private createTimeAxisOptions(endTime: string | undefined): EChartsCoreOption['xAxis'] {
    return {
      type: 'time',
      boundaryGap: false,
      max: endTime,
      axisLine: {
        lineStyle: {
          color: this.chartTheme.axis,
        },
      },
      axisLabel: {
        color: this.chartTheme.textSecondary,
      },
      splitLine: {
        lineStyle: {
          color: this.chartTheme.grid,
        },
      },
    };
  }

  private createValueAxisOptions(): EChartsCoreOption['yAxis'] {
    return {
      type: 'value',
      min: 0,
      axisLine: {
        lineStyle: {
          color: this.chartTheme.axis,
        },
      },
      axisLabel: {
        color: this.chartTheme.textSecondary,
      },
      splitLine: {
        lineStyle: {
          color: this.chartTheme.grid,
        },
      },
    };
  }

  private createSeries(endTime: string | undefined): EChartsCoreOption['series'] {
    return this.teams.map((team) => ({
      name: team.name,
      type: 'line',
      step: 'end',
      showSymbol: false,
      lineStyle: {
        width: 2,
      },
      emphasis: {
        focus: 'series',
      },
      data: this.getCumulativeScoreData(team.solves, endTime),
    }));
  }

  private getLastSolveTime(): string | undefined {
    const timestamps = this.teams
      .flatMap((team) => team.solves)
      .map((solve) => new Date(solve.date).getTime())
      .filter((timestamp) => Number.isFinite(timestamp));

    if (!timestamps.length) {
      return undefined;
    }

    return new Date(Math.max(...timestamps)).toISOString();
  }

  private getCumulativeScoreData(
    solves: ScoreboardChartSolve[],
    endTime: string | undefined
  ): [string, number][] {
    let score = 0;

    const data: [string, number][] = [...solves]
      .filter((solve) => solve.date)
      .sort((a, b) => new Date(a.date).getTime() - new Date(b.date).getTime())
      .map((solve) => {
        score += Number(solve.value) || 0;

        return [solve.date, score] as [string, number];
      });

    const lastPoint = data.at(-1);

    if (lastPoint && endTime && new Date(lastPoint[0]).getTime() < new Date(endTime).getTime()) {
      data.push([endTime, lastPoint[1]]);
    }

    return data;
  }
}

export { ScoreboardChart as Scoreboard };

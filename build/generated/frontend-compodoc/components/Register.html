import {Component, inject, signal} from '@angular/core';
import {HttpErrorResponse} from '@angular/common/http';
import {ButtonModule, ContextMenuModule, InputModule, NotificationModule} from "carbon-components-angular";
import {FormControl, FormGroup, ReactiveFormsModule} from "@angular/forms";
import {TranslocoDirective, TranslocoService} from "@ngneat/transloco";
import {ClientConfig} from '../../../shared/config/client.config';
import {AuthApi} from '../../../utils/auth/auth-api';
import {Router, RouterLink} from '@angular/router';
import {catchError, of} from 'rxjs';
import {RegisterInterface} from '../../../entities/auth/register.interface';
import {controlInvalid, getControlErrorKey, registerValidators} from '../../../utils/auth/auth-form-validators';

@Component({
  selector: 'app-register',
  imports: [
    ButtonModule,
    ContextMenuModule,
    InputModule,
    NotificationModule,
    ReactiveFormsModule,
    TranslocoDirective,
    RouterLink
  ],
  templateUrl: './register.html',
  styleUrl: './register.scss',
})
export class Register {
  clientConfig = inject(ClientConfig);
  private readonly authApi = inject(AuthApi);
  private readonly router = inject(Router);
  private readonly translocoService = inject(TranslocoService);

  supportContact = signal<string>(this.clientConfig.getSupportContact());
  notification = signal<{ title: string, message: string } | null>(null);

  form = new FormGroup({
    email: new FormControl('', registerValidators.email),
    username: new FormControl('', registerValidators.username),
    password: new FormControl('', registerValidators.password),
  });

  isInvalid(controlName: 'email' | 'username' | 'password') {
    return controlInvalid(this.form.get(controlName));
  }

  getErrorMessage(controlName: 'email' | 'username' | 'password') {
    const errorKey = getControlErrorKey(this.form.get(controlName));
    return errorKey ? this.translocoService.translate(`auth-pages.validation.${controlName}.${errorKey}`) : '';
  }

  onSubmit() {
    if (this.form.valid) {
      this.authApi.register(this.form.value as RegisterInterface)
        .pipe(
          catchError((error: HttpErrorResponse) => {
            if (error.status === 401) {
              this.notification.set({
                title: this.translocoService.translate('auth-pages.error.title'),
                message: this.translocoService.translate('auth-pages.error.message-invalid'),
              });
            } else if (error.status === 0) {
              this.notification.set({
                title: this.translocoService.translate('auth-pages.error.title'),
                message: this.translocoService.translate('auth-pages.error.message-no-connection')
              });
            } else {
              this.notification.set({
                title: this.translocoService.translate('auth-pages.error.title'),
                message: this.translocoService.translate('auth-pages.error.message-universal')
              });
            }
            return of(null);
          })
        )
        .subscribe(res => {
          if (res) {
            void this.router.navigate(['']);
          }
        });
    } else {
      this.form.markAllAsTouched();
    }
  }
}

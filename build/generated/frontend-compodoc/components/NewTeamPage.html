import {Component, inject, signal} from '@angular/core';
import {HttpErrorResponse} from '@angular/common/http';
import {FormControl, FormGroup, ReactiveFormsModule, Validators} from '@angular/forms';
import {Router} from '@angular/router';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  ContextMenuModule,
  InputModule,
  NotificationModule,
  SelectModule
} from 'carbon-components-angular';
import {catchError, finalize, of} from 'rxjs';
import {CreateTeamInterface, JoinTeamInterface} from '../../entities/team.interface';
import {TeamApi} from '../../utils/team/team-api';
import {controlInvalid, getControlErrorKey} from '../../utils/auth/auth-form-validators';

type CreateControlName = 'name' | 'password' | 'category' | 'affiliation';
type JoinControlName = 'team_slug' | 'password';

const TEAM_NAME_MAX_LENGTH = 64;
const TEAM_SLUG_PATTERN = /^[a-z0-9]+(?:-[a-z0-9]+)*$/;
const TEAM_PASSWORD_MIN_LENGTH = 8;

@Component({
  selector: 'app-new-team-page',
  imports: [
    ButtonModule,
    ContextMenuModule,
    InputModule,
    NotificationModule,
    ReactiveFormsModule,
    SelectModule,
    TranslocoDirective
  ],
  templateUrl: './new-team-page.html',
  styleUrl: './new-team-page.scss',
})
export class NewTeamPage {
  private readonly router = inject(Router);
  private readonly teamApi = inject(TeamApi);
  private readonly transloco = inject(TranslocoService);

  readonly mode = signal<'create' | 'join'>('create');
  readonly loading = signal<boolean>(false);
  readonly notification = signal<{ title: string, message: string } | null>(null);

  readonly categories = ['school', 'university', 'company', 'open'];

  readonly createForm = new FormGroup({
    name: new FormControl('', [
      Validators.required,
      Validators.maxLength(TEAM_NAME_MAX_LENGTH),
    ]),
    password: new FormControl('', [
      Validators.required,
      Validators.minLength(TEAM_PASSWORD_MIN_LENGTH),
    ])
  });

  readonly joinForm = new FormGroup({
    team_slug: new FormControl('', [
      Validators.required,
      Validators.pattern(TEAM_SLUG_PATTERN),
    ]),
    password: new FormControl('', [
      Validators.required,
      Validators.minLength(TEAM_PASSWORD_MIN_LENGTH),
    ]),
  });

  setMode(mode: 'create' | 'join') {
    this.mode.set(mode);
    this.notification.set(null);
  }

  isCreateInvalid(controlName: CreateControlName) {
    return controlInvalid(this.createForm.get(controlName));
  }

  isJoinInvalid(controlName: JoinControlName) {
    return controlInvalid(this.joinForm.get(controlName));
  }

  getCreateErrorMessage(controlName: CreateControlName) {
    return this.getErrorMessage('create', controlName, this.createForm);
  }

  getJoinErrorMessage(controlName: JoinControlName) {
    return this.getErrorMessage('join', controlName, this.joinForm);
  }

  onCreateSubmit() {
    if (this.createForm.invalid) {
      this.createForm.markAllAsTouched();
      return;
    }

    this.loading.set(true);
    this.notification.set(null);

    this.teamApi.createTeam(this.createForm.getRawValue() as CreateTeamInterface)
      .pipe(
        catchError((error: HttpErrorResponse) => {
          this.showRequestError(error.status, 'create');
          return of(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => {
        if (res?.team) {
          void this.router.navigate(['/team', res.team.id]);
        }
      });
  }

  onJoinSubmit() {
    if (this.joinForm.invalid) {
      this.joinForm.markAllAsTouched();
      return;
    }

    this.loading.set(true);
    this.notification.set(null);

    this.teamApi.joinTeam(this.joinForm.getRawValue() as JoinTeamInterface)
      .pipe(
        catchError((error: HttpErrorResponse) => {
          this.showRequestError(error.status, 'join');
          return of(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => {
        if (res?.team) {
          void this.router.navigate(['/team', res.team.id]);
        }
      });
  }

  private getErrorMessage(
    formName: 'create' | 'join',
    controlName: CreateControlName | JoinControlName,
    form: FormGroup
  ) {
    const errorKey = getControlErrorKey(form.get(controlName));

    return errorKey ? this.transloco.translate(`new-team-page.validation.${formName}.${controlName}.${errorKey}`) : '';
  }

  private showRequestError(status: number, formName: 'create' | 'join') {
    const messageKey = status === 0
      ? 'no-connection'
      : status === 400 || status === 401 || status === 403 || status === 404 || status === 409
        ? `${formName}-invalid`
        : 'universal';

    this.notification.set({
      title: this.transloco.translate('new-team-page.error.title'),
      message: this.transloco.translate(`new-team-page.error.${messageKey}`),
    });
  }
}

import {Component, inject, signal} from '@angular/core';
import {CommonModule} from '@angular/common';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  LoadingModule,
  TableHeaderItem,
  TableItem,
  TableModel,
  TableModule,
} from 'carbon-components-angular';
import {AdminApi} from '../../utils/api/admin-api';
import {AdminTeam} from '../../entities/admin/admin.interface';
import {matchesSearch} from '../../utils/search/table-search';

@Component({
  selector: 'app-admin-teams',
  imports: [
    CommonModule,
    TranslocoDirective,
    TableModule,
    LoadingModule,
    ButtonModule,
  ],
  templateUrl: './admin-teams.html',
  styleUrl: './admin-teams.scss',
})
export class AdminTeams {
  private readonly adminApi = inject(AdminApi);
  private readonly transloco = inject(TranslocoService);

  readonly loading = signal(true);
  private allTeams: AdminTeam[] = [];
  private searchQuery = '';
  teams = this.createModel([]);

  constructor() {
    this.loadData();
  }

  private loadData() {
    this.loading.set(true);
    this.adminApi.getAdminTeams().subscribe((res) => {
      this.allTeams = res.teams;
      this.applySearch();
      this.loading.set(false);
    });
  }

  private createModel(items: AdminTeam[]) {
    const model = new TableModel();
    model.header = [
      new TableHeaderItem({ data: this.transloco.translate('admin.teams.columns.name') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.teams.columns.slug') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.teams.columns.category') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.teams.columns.affiliation') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.teams.columns.score') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.teams.columns.members') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.teams.columns.solves') }),
    ];
    model.data = items.map((t) => [
      new TableItem({ data: t.name }),
      new TableItem({ data: t.slug }),
      new TableItem({ data: t.category }),
      new TableItem({ data: t.affiliation }),
      new TableItem({ data: t.score }),
      new TableItem({ data: t.members_count }),
      new TableItem({ data: t.solves_count }),
    ]);
    return model;
  }

  refresh() {
    this.loadData();
  }

  onSearch(query: string) {
    this.searchQuery = query;
    this.applySearch();
  }

  private applySearch() {
    const filtered = this.allTeams.filter((team) => matchesSearch([
      team.id,
      team.name,
      team.slug,
      team.category,
      team.affiliation,
      team.score,
      team.members_count,
      team.solves_count,
      team.created_at,
      team.last_solve,
    ], this.searchQuery));

    this.teams = this.createModel(filtered);
  }
}

<ng-container *transloco="let t">
  <div class="container flex flex-col gap-[1.5rem]">
    <div class="flex items-center justify-between">
      <h1 class="cds--type-heading-07">{{ t('admin.teams.title') }}</h1>
      <button cdsButton type="button" (click)="refresh()">{{ t('admin.teams.refresh') }}</button>
    </div>

    @if (loading()) {
      <cds-loading [isActive]="true" size="normal"></cds-loading>

          <cds-table-toolbar-content>
            <cds-table-toolbar-search
              [expandable]="true"
              (valueChange)="onSearch($event)">
            </cds-table-toolbar-search>
          </cds-table-toolbar-content>
        </cds-table-toolbar>

        <cds-table
          [model]="teams"
          size="md"
          [showSelectionColumn]="false"

        </cds-table>

import {inject, Injectable} from '@angular/core';
import {HttpClient} from '@angular/common/http';
import {ApiConfig} from '../../shared/config/api.config';
import {
  CreateTeamInterface,
  JoinTeamInterface,
  LeaderboardResponseInterface,
  TeamMembersInterface,
  TeamDashboardInterface,
  TeamResponseInterface,
  TeamsResponseInterface,
  TeamSolvesResponseInterface,
  UpdateTeamInterface
} from '../../entities/team.interface';

@Injectable({
  providedIn: 'root',
})
export class TeamApi {
  private readonly http = inject(HttpClient);
  private readonly baseApiUrl = inject(ApiConfig).getBaseApiUrl();

  getTeam(teamId: string | null = null) {
    if (teamId) {
      return this.http.get<TeamResponseInterface>(`${this.baseApiUrl}teams/${teamId}`,
        {
          withCredentials: true,
        }
      );
    }

    return this.http.get<TeamResponseInterface>(`${this.baseApiUrl}team`,
      {
        withCredentials: true,
      }
    );
  }

  getTeamMembers(teamId: string | null = null) {
    if (teamId) {
      return this.http.get<TeamMembersInterface>(`${this.baseApiUrl}teams/${teamId}/members`,
        {
          withCredentials: true,
        }
      );
    }

    return this.http.get<TeamMembersInterface>(`${this.baseApiUrl}team/members`,
      {
        withCredentials: true,
      }
    );
  }

  getTeamSolves(teamId: string | null = null, limit?: number) {
    if (teamId) {
      return this.http.get<TeamSolvesResponseInterface>(`${this.baseApiUrl}teams/${teamId}/solves`,
        {
          withCredentials: true,
        }
      );
    }

    return this.http.get<TeamSolvesResponseInterface>(`${this.baseApiUrl}team/solves`,
      {
        params: limit ? { limit } : undefined,
        withCredentials: true,
      }
    );
  }

  createTeam(payload: CreateTeamInterface) {
    return this.http.post<TeamResponseInterface>(
      `${this.baseApiUrl}team`,
      payload,
      {
        withCredentials: true,
      }
    );
  }

  joinTeam(payload: JoinTeamInterface) {
    return this.http.post<TeamResponseInterface>(
      `${this.baseApiUrl}team/join`,
      payload,
      {
        withCredentials: true,
      }
    );
  }

  updateTeam(payload: UpdateTeamInterface) {
    return this.http.put<TeamResponseInterface>(
      `${this.baseApiUrl}team`,
      payload,
      {
        withCredentials: true,
      }
    );
  }

  deleteTeam() {
    return this.http.delete<Record<string, never>>(
      `${this.baseApiUrl}team`,
      {
        withCredentials: true,
      }
    );
  }

  deleteTeamMember(userId: string) {
    return this.http.delete<Record<string, never>>(
      `${this.baseApiUrl}team/members/${userId}`,
      {
        withCredentials: true,
      }
    );
  }

  getTeamDashboard() {
    return this.http.get<TeamDashboardInterface>(`${this.baseApiUrl}team/dashboard`,
      {
        withCredentials: true,
      }
    );
  }

  getTeams() {
    return this.http.get<TeamsResponseInterface>(`${this.baseApiUrl}teams`,
      {
        withCredentials: true,
      }
    );
  }

  getTeamMembersById(teamId: string) {
    return this.getTeamMembers(teamId);
  }

  getTeamSolvesById(teamId: string) {
    return this.getTeamSolves(teamId);
  }

  getLeaderboard() {
    return this.http.get<LeaderboardResponseInterface>(`${this.baseApiUrl}leaderboard`,
      {
        withCredentials: true,
      }
    );
  }
}

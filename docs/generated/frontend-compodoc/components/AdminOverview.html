import {Component, computed, inject, signal} from '@angular/core';
import {CommonModule} from '@angular/common';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  LoadingModule,
  TableHeaderItem,
  TableItem,
  TableModel,
  TableModule,
  TilesModule,
} from 'carbon-components-angular';
import {AdminApi} from '../../utils/api/admin-api';
import {AdminOverviewResponse, AdminSolve, AdminInstance} from '../../entities/admin/admin.interface';
import {matchesSearch} from '../../utils/search/table-search';

@Component({
  selector: 'app-admin-overview',
  imports: [
    CommonModule,
    TranslocoDirective,
    TableModule,
    LoadingModule,
    TilesModule,
    ButtonModule,
  ],
  templateUrl: './admin-overview.html',
  styleUrl: './admin-overview.scss',
})
export class AdminOverview {
  private readonly adminApi = inject(AdminApi);
  private readonly transloco = inject(TranslocoService);

  readonly loading = signal(true);
  overview = signal<AdminOverviewResponse | null>(null);
  topTeamsQuery = signal('');
  recentSolvesQuery = signal('');
  activeInstancesQuery = signal('');

  statsList = computed(() => {
    const o = this.overview();
    if (!o) return [];
    return [
      { label: this.transloco.translate('admin.overview.stats.users'), value: o.stats.users },
      { label: this.transloco.translate('admin.overview.stats.teams'), value: o.stats.teams },
      { label: this.transloco.translate('admin.overview.stats.challenges'), value: o.stats.challenges },
      { label: this.transloco.translate('admin.overview.stats.visible'), value: o.stats.visible_challenges },
      { label: this.transloco.translate('admin.overview.stats.hidden'), value: o.stats.hidden_challenges },
      { label: this.transloco.translate('admin.overview.stats.dynamic'), value: o.stats.dynamic_challenges },
      { label: this.transloco.translate('admin.overview.stats.solves'), value: o.stats.solves },
      { label: this.transloco.translate('admin.overview.stats.instances'), value: o.stats.active_instances },
    ];
  });

  topTeamsModel = computed(() => {
    const model = new TableModel();
    model.header = [
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.team') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.score') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.members') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.solves') }),
    ];
    const teams = this.overview()?.top_teams.filter(t => matchesSearch([
      t.id,
      t.name,
      t.slug,
      t.category,
      t.affiliation,
      t.score,
      t.members_count,
      t.solves_count,
      t.created_at,
      t.last_solve,
    ], this.topTeamsQuery())) ?? [];

    model.data = teams.map(t => [
      new TableItem({ data: t.name }),
      new TableItem({ data: t.score }),
      new TableItem({ data: t.members_count }),
      new TableItem({ data: t.solves_count }),
    ]);
    return model;
  });

  recentSolvesModel = computed(() => {
    const model = new TableModel();
    model.header = [
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.challenge') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.team') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.points') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.solvedAt') }),
    ];
    const solves = this.overview()?.recent_solves.filter((s: AdminSolve) => matchesSearch([
      s.id,
      s.challenge_id,
      s.challenge_title,
      s.category,
      s.team_id,
      s.team_name,
      s.team_slug,
      s.user_id,
      s.username,
      s.points,
      s.solved_at,
      s.is_first_blood,
    ], this.recentSolvesQuery())) ?? [];

    model.data = solves.map((s: AdminSolve) => [
      new TableItem({ data: s.challenge_title }),
      new TableItem({ data: s.team_name }),
      new TableItem({ data: s.points }),
      new TableItem({ data: s.solved_at }),
    ]);
    return model;
  });

  activeInstancesModel = computed(() => {
    const model = new TableModel();
    model.header = [
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.instanceChallenge') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.instanceTeam') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.instanceStatus') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.overview.columns.instanceExpires') }),
    ];
    const instances = this.overview()?.active_instances.filter((i: AdminInstance) => matchesSearch([
      i.id,
      i.challenge_id,
      i.challenge_title,
      i.team_id,
      i.team_name,
      i.instance_name,
      i.connection_url,
      i.status,
      i.error_message,
      i.expires_at,
      i.created_at,
      i.updated_at,
    ], this.activeInstancesQuery())) ?? [];

    model.data = instances.map((i: AdminInstance) => [
      new TableItem({ data: i.challenge_title }),
      new TableItem({ data: i.team_name }),
      new TableItem({ data: i.status }),
      new TableItem({ data: i.expires_at }),
    ]);
    return model;
  });

  constructor() {
    this.adminApi.getOverview().subscribe((res) => {
      this.overview.set(res);
      this.loading.set(false);
    });
  }
}

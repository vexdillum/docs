import {AfterViewInit, Component, inject, signal, TemplateRef, ViewChild} from '@angular/core';
import {CommonModule} from '@angular/common';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  LoadingModule,
  ModalModule,
  TableHeaderItem,
  TableItem,
  TableModel,
  TableModule,
} from 'carbon-components-angular';
import {AdminApi} from '../../utils/api/admin-api';
import {AdminUser} from '../../entities/admin/admin.interface';
import {matchesSearch} from '../../utils/search/table-search';

@Component({
  selector: 'app-admin-users',
  imports: [
    CommonModule,
    TranslocoDirective,
    TableModule,
    LoadingModule,
    ButtonModule,
    ModalModule,
  ],
  templateUrl: './admin-users.html',
  styleUrl: './admin-users.scss',
})
export class AdminUsers implements AfterViewInit {
  @ViewChild('linkTemplate')
  protected linkTemplate!: TemplateRef<{ data: AdminUser }>;

  private readonly adminApi = inject(AdminApi);
  private readonly transloco = inject(TranslocoService);

  readonly loading = signal(true);
  private allUsers: AdminUser[] = [];
  private searchQuery = '';
  users = this.createModel([]);

  selectedUser = signal<AdminUser | null>(null);
  selectedDetailUser = signal<AdminUser | null>(null);
  openDeleteModal = signal(false);
  openDetailModal = signal(false);

  ngAfterViewInit() {
    this.loadData();
  }

  private loadData() {
    this.loading.set(true);
    this.adminApi.getAdminUsers().subscribe((res) => {
      this.allUsers = res.users;
      this.applySearch();
      this.loading.set(false);
    });
  }

  private createModel(items: AdminUser[]) {
    const model = new TableModel();
    model.header = [
      new TableHeaderItem({ data: this.transloco.translate('admin.users.columns.username') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.users.columns.email') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.users.columns.role') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.users.columns.team') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.users.columns.banned') }),
    ];
    model.data = items.map((u) => [
      new TableItem({
        data: u,
        template: this.linkTemplate,
      }),
      new TableItem({ data: u.email }),
      new TableItem({ data: u.role }),
      new TableItem({ data: u.team_name ?? '-' }),
      new TableItem({ data: u.is_banned ? 'Yes' : 'No' }),
    ]);
    return model;
  }

  refresh() {
    this.loadData();
  }

  onSearch(query: string) {
    this.searchQuery = query;
    this.applySearch();
  }

  openDetail(user: AdminUser) {
    this.selectedDetailUser.set(user);
    this.openDetailModal.set(true);
  }

  ban(user: AdminUser) {
    this.adminApi.banUser(user.id).subscribe(() => {
      this.loadData();
      this.selectedDetailUser.update(u => u?.id === user.id ? { ...u, is_banned: true } : u);
    });
  }

  unban(user: AdminUser) {
    this.adminApi.unbanUser(user.id).subscribe(() => {
      this.loadData();
      this.selectedDetailUser.update(u => u?.id === user.id ? { ...u, is_banned: false } : u);
    });
  }

  askDelete(user: AdminUser) {
    this.selectedUser.set(user);
    this.openDeleteModal.set(true);
  }

  askDeleteFromDetail(user: AdminUser) {
    this.openDetailModal.set(false);
    this.askDelete(user);
  }

  confirmDelete() {
    const id = this.selectedUser()?.id;
    if (!id) return;
    this.adminApi.deleteAdminUser(id, { confirm: 'DELETE_USER' }).subscribe(() => {
      this.openDeleteModal.set(false);
      this.selectedUser.set(null);
      this.loadData();
    });
  }

  private applySearch() {
    const filtered = this.allUsers.filter((user) => matchesSearch([
      user.id,
      user.username,
      user.email,
      user.role,
      user.team_id,
      user.team_name,
      user.team_slug,
      user.is_captain,
      user.is_banned,
      user.is_captain ? 'Yes' : 'No',
      user.is_banned ? 'Yes' : 'No',
      user.points_contributed,
      user.solves_count,
      user.created_at,
    ], this.searchQuery));

    this.users = this.createModel(filtered);
  }
}

import {AfterViewInit, Component, inject, signal, TemplateRef, ViewChild} from '@angular/core';
import {RouterLink} from '@angular/router';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {TeamApi} from '../../utils/team/team-api';
import {TableHeaderItem, TableItem, TableModel, TableModule} from 'carbon-components-angular';
import {Location} from '@angular/common';
import {catchError, finalize, of} from 'rxjs';
import {TeamInterface} from '../../entities/team.interface';

interface TeamTableItem {
  name: string;
  link: string[];
}

@Component({
  selector: 'app-teams-list-page',
  imports: [
    RouterLink,
    TranslocoDirective,
    TableModule
  ],
  templateUrl: './teams-list-page.html',
  styleUrl: './teams-list-page.scss',
})
export class TeamsListPage implements AfterViewInit {
  @ViewChild('teamCellTemplate')
  protected teamCellTemplate!: TemplateRef<{ data: TeamTableItem }>;

  private readonly teamApi = inject(TeamApi);
  private readonly location = inject(Location);
  private readonly transloco = inject(TranslocoService);

  readonly loading = signal<boolean>(true);

  private allTeams: TeamInterface[] = [];
  teams = this.createTeamsModel([]);

  ngAfterViewInit() {
    this.loadData();
  }

  private loadData() {
    this.loading.set(true);
    this.teams = this.createTeamsModel([]);

    this.teamApi.getTeams()
      .pipe(
        catchError(() => {
          this.location.back();
          return of(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => {
        this.allTeams = res?.teams ?? [];
        this.teams = this.createTeamsModel(this.allTeams);
      });
  }

  onSearch(query: string): void {
    const normalizedQuery = query.trim().toLowerCase();
    const filtered = normalizedQuery
      ? this.allTeams.filter((team) => {
          const fields = [
            team.name,
            team.slug,
            team.category,
            team.affiliation,
            team.score?.toString(),
          ];
          return fields.some((field) => field?.toLowerCase().includes(normalizedQuery));
        })
      : this.allTeams;

    this.teams = this.createTeamsModel(filtered);
  }

  private createTeamsModel(teams: TeamInterface[]) {
    const model = new TableModel();

    model.header = [
      new TableHeaderItem({data: this.transloco.translate('teams-list-page.columns.name')}),
      new TableHeaderItem({data: this.transloco.translate('teams-list-page.columns.affiliation')}),
      new TableHeaderItem({data: this.transloco.translate('teams-list-page.columns.score')})
    ];

    model.data = teams.map((team) => [
      new TableItem({
        data: {
          name: team.name,
          link: ['/team', team.id],
        },
        template: this.teamCellTemplate
      }),
      new TableItem({data: team.affiliation}),
      new TableItem({data: team.score}),
    ]);

    return model;
  }
}

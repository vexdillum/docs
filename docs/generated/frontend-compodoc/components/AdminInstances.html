import {AfterViewInit, Component, inject, signal, TemplateRef, ViewChild} from '@angular/core';
import {CommonModule} from '@angular/common';
import {FormControl, FormGroup, ReactiveFormsModule, Validators} from '@angular/forms';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  InputModule,
  LoadingModule,
  ModalModule,
  NotificationModule,
  TagModule,
  TableHeaderItem,
  TableItem,
  TableModel,
  TableModule,
} from 'carbon-components-angular';
import {AdminApi} from '../../utils/api/admin-api';
import {AdminInstance} from '../../entities/admin/admin.interface';
import {catchError, finalize, of} from 'rxjs';
import {getInstanceStatusView} from '../../utils/instances/instance-status';
import {matchesSearch} from '../../utils/search/table-search';

@Component({
  selector: 'app-admin-instances',
  imports: [
    CommonModule,
    TranslocoDirective,
    TableModule,
    LoadingModule,
    ButtonModule,
    ModalModule,
    NotificationModule,
    TagModule,
    InputModule,
    ReactiveFormsModule,
  ],
  templateUrl: './admin-instances.html',
  styleUrl: './admin-instances.scss',
})
export class AdminInstances implements AfterViewInit {
  @ViewChild('linkTemplate')
  protected linkTemplate!: TemplateRef<{ data: AdminInstance }>;

  @ViewChild('statusTemplate')
  protected statusTemplate!: TemplateRef<{ data: ReturnType<typeof getInstanceStatusView> }>;

  @ViewChild('connectionTemplate')
  protected connectionTemplate!: TemplateRef<{ data: string }>;

  @ViewChild('actionsTemplate')
  protected actionsTemplate!: TemplateRef<{ data: { instance: AdminInstance } }>;

  private readonly adminApi = inject(AdminApi);
  private readonly transloco = inject(TranslocoService);

  readonly loading = signal(true);
  readonly error = signal<string | null>(null);
  readonly hasInstances = signal(false);
  readonly detailLoading = signal(false);
  readonly detailError = signal<string | null>(null);
  readonly mutationLoading = signal<string | null>(null);
  readonly notification = signal<{ type: 'error' | 'success'; title: string; message: string } | null>(null);
  private allInstances: AdminInstance[] = [];
  private searchQuery = '';
  private detailInstanceId: string | null = null;
  instances = this.createModel([]);

  selectedInstance = signal<AdminInstance | null>(null);
  selectedDetailInstance = signal<AdminInstance | null>(null);
  openDeleteModal = signal(false);
  openDetailModal = signal(false);

  statusForm = new FormGroup({
    status: new FormControl('', [Validators.required]),
    error_message: new FormControl(''),
  });

  ngAfterViewInit() {
    this.loadData();
  }

  private loadData() {
    this.loading.set(true);
    this.error.set(null);

    this.adminApi.getInstances()
      .pipe(
        catchError(() => {
          this.error.set(this.transloco.translate('admin.instances.errors.load'));
          return of({ instances: [] });
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => {
        this.allInstances = res.instances;
        this.hasInstances.set(this.allInstances.length > 0);
        this.applySearch();
      });
  }

  private createModel(items: AdminInstance[]) {
    const model = new TableModel();
    model.header = [
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.challenge') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.team') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.status') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.connection') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.expires') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.created') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.updated') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.instances.columns.actions') }),
    ];
    model.data = items.map((i) => [
      new TableItem({
        data: i,
        template: this.linkTemplate,
      }),
      new TableItem({ data: i.team_name }),
      new TableItem({
        data: this.getStatusView(i),
        template: this.statusTemplate,
      }),
      new TableItem({
        data: i.connection_url,
        template: this.connectionTemplate,
      }),
      new TableItem({ data: this.formatDate(i.expires_at) }),
      new TableItem({ data: this.formatDate(i.created_at) }),
      new TableItem({ data: this.formatDate(i.updated_at) }),
      new TableItem({
        data: { instance: i },
        template: this.actionsTemplate,
      }),
    ]);
    return model;
  }

  getStatusView(instance: AdminInstance) {
    return getInstanceStatusView(instance.status, instance.expires_at);
  }

  refresh() {
    this.loadData();
  }

  onSearch(query: string) {
    this.searchQuery = query;
    this.applySearch();
  }

  openDetail(instance: AdminInstance) {
    const requestedInstanceId = instance.id;

    this.detailInstanceId = requestedInstanceId;
    this.openDetailModal.set(true);
    this.selectedDetailInstance.set(null);
    this.detailLoading.set(true);
    this.detailError.set(null);
    this.notification.set(null);

    this.adminApi.getInstance(instance.id)
      .pipe(
        catchError(() => {
          if (this.detailInstanceId === requestedInstanceId) {
            this.detailError.set(this.transloco.translate('admin.instances.errors.detail'));
          }
          return of(null);
        })
      )
      .subscribe((res) => {
        if (this.detailInstanceId !== requestedInstanceId) {
          return;
        }

        this.detailLoading.set(false);

        if (!res) {
          return;
        }

        this.setDetailInstance(res.instance);
      });
  }

  extend(instance: AdminInstance) {
    this.mutateInstance(
      instance.id,
      this.adminApi.extendInstance(instance.id, { ttl_seconds: 3600 }),
      'extendSuccess',
      'extendFailed'
    );
  }

  restart(instance: AdminInstance) {
    this.mutateInstance(
      instance.id,
      this.adminApi.restartInstance(instance.id, { ttl_seconds: 3600 }),
      'restartSuccess',
      'restartFailed'
    );
  }

  updateStatus() {
    const instance = this.selectedDetailInstance();

    if (!instance || this.statusForm.invalid) {
      this.statusForm.markAllAsTouched();
      return;
    }

    const raw = this.statusForm.value;
    const errorMessage = raw.error_message?.trim();

    this.mutateInstance(
      instance.id,
      this.adminApi.updateInstanceStatus(instance.id, {
        status: raw.status!,
        error_message: errorMessage === '' ? undefined : errorMessage,
      }),
      'statusSuccess',
      'statusFailed'
    );
  }

  askDelete(instance: AdminInstance) {
    this.selectedInstance.set(instance);
    this.openDeleteModal.set(true);
  }

  askDeleteFromDetail(instance: AdminInstance) {
    this.openDetailModal.set(false);
    this.askDelete(instance);
  }

  confirmDelete() {
    const id = this.selectedInstance()?.id;
    if (!id) return;
    this.mutationLoading.set(id);
    this.notification.set(null);

    this.adminApi.deleteInstance(id, { confirm: 'DELETE_INSTANCE' })
      .pipe(
        catchError(() => {
          this.showNotification('error', 'deleteFailed');
          return of(null);
        }),
        finalize(() => this.mutationLoading.set(null))
      )
      .subscribe((res) => {
        if (!res) {
          return;
        }

        this.openDeleteModal.set(false);
        this.selectedInstance.set(null);

        if (this.selectedDetailInstance()?.id === id) {
          this.selectedDetailInstance.set(null);
          this.openDetailModal.set(false);
        }

        this.showNotification('success', 'deleteSuccess');
        this.loadData();
      });
  }

  private mutateInstance(
    instanceId: string,
    request: ReturnType<AdminApi['extendInstance']>,
    successKey: string,
    errorKey: string
  ) {
    this.mutationLoading.set(instanceId);
    this.notification.set(null);

    request
      .pipe(
        catchError(() => {
          this.showNotification('error', errorKey);
          return of(null);
        }),
        finalize(() => this.mutationLoading.set(null))
      )
      .subscribe((res) => {
        if (!res) {
          return;
        }

        this.setDetailInstance(res.instance);
        this.showNotification('success', successKey);
        this.loadData();
      });
  }

  private setDetailInstance(instance: AdminInstance) {
    this.selectedDetailInstance.set(instance);
    this.statusForm.reset({
      status: instance.status,
      error_message: instance.error_message ?? '',
    });
  }

  private showNotification(type: 'error' | 'success', key: string) {
    this.notification.set({
      type,
      title: this.transloco.translate(`admin.instances.notifications.${key}.title`),
      message: this.transloco.translate(`admin.instances.notifications.${key}.message`),
    });
  }

  private applySearch() {
    const filtered = this.allInstances.filter((instance) => matchesSearch([
      instance.id,
      instance.challenge_id,
      instance.challenge_title,
      instance.team_id,
      instance.team_name,
      instance.instance_name,
      instance.connection_url,
      instance.status,
      instance.error_message,
      instance.expires_at,
      instance.created_at,
      instance.updated_at,
      this.formatDate(instance.expires_at),
      this.formatDate(instance.created_at),
      this.formatDate(instance.updated_at),
    ], this.searchQuery));

    this.instances = this.createModel(filtered);
  }

  private formatDate(value: string): string {
    const date = new Date(value);

    if (Number.isNaN(date.getTime())) {
      return value;
    }

    return date.toLocaleString(this.transloco.getActiveLang());
  }
}

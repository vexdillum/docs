import {
  AfterViewInit,
  Component,
  inject,
  Input,
  OnChanges,
  signal,
  SimpleChanges,
  TemplateRef,
  ViewChild
} from '@angular/core';
import {DatePipe, Location} from '@angular/common';
import {RouterLink} from '@angular/router';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  TagModule,
  TableModel,
  TableModule
} from 'carbon-components-angular';
import {catchError, finalize, forkJoin, map, Observable, of, switchMap} from 'rxjs';
import {TeamInterface, TeamSolveInterface} from '../../entities/team.interface';
import {UserInterface} from '../../entities/user.interface';
import {UsersApi} from '../../utils/api/users';
import {TeamApi} from '../../utils/team/team-api';
import {MeApi} from '../../utils/auth/me-api';
import {EditMeModal} from '../../widgets/edit-me-modal/edit-me-modal';
import {MeInterface} from '../../entities/auth/me.interface';
import {
  createMembersModel,
  createSolvesModel,
  getUserBadges,
  MemberBadge,
  MemberTableItem
} from '../../utils/team/team-table-models';

interface UserSeed {
  user: UserInterface | null;
  team: TeamInterface | null;
  currentTeam: TeamInterface | null;
  me: UserInterface | null;
}

interface UserTeam {
  team: TeamInterface | null;
  members: UserInterface[];
}

interface UserPageData {
  user: UserInterface | null;
  team: TeamInterface | null;
  members: UserInterface[];
  solves: TeamSolveInterface[];
  me: UserInterface | null;
}

@Component({
  selector: 'app-user-page',
  imports: [
    ButtonModule,
    RouterLink,
    TranslocoDirective,
    TagModule,
    TableModule,
    EditMeModal
  ],
  templateUrl: './user-page.html',
  styleUrl: '../team-page/team-page.scss',
  providers: [DatePipe],
})
export class UserPage implements AfterViewInit, OnChanges {
  @Input() user_id?: string;
  @ViewChild('memberCellTemplate')
  protected memberCellTemplate!: TemplateRef<{ data: MemberTableItem }>;

  private readonly usersApi = inject(UsersApi);
  private readonly teamApi = inject(TeamApi);
  private readonly meApi = inject(MeApi);
  private readonly datePipe = inject(DatePipe);
  private readonly location = inject(Location);
  private readonly transloco = inject(TranslocoService);

  readonly user = signal<UserInterface | null>(null);
  readonly team = signal<TeamInterface | null>(null);
  readonly membersList = signal<UserInterface[] | null>(null);
  readonly solvesList = signal<TeamSolveInterface[] | null>(null);
  readonly myUser = signal<UserInterface | null>(null);
  readonly loading = signal<boolean>(true);
  readonly isMyUser = signal<boolean>(false);
  readonly editMeModalOpen = signal<boolean>(false);

  members = new TableModel();
  solves = new TableModel();

  ngAfterViewInit(): void {
    this.loadPage();
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes['user_id'] && !changes['user_id'].firstChange) {
      this.loadPage();
    }
  }

  private loadPage() {
    const userId = this.user_id ?? null;

    this.resetPageState();
    this.fetchUserPageData(userId)
      .pipe(
        catchError(() => {
          this.location.back();
          return of<UserPageData | null>(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => this.applyPageData(res));
  }

  private resetPageState(): void {
    this.loading.set(true);
    this.user.set(null);
    this.team.set(null);
    this.membersList.set(null);
    this.solvesList.set(null);
    this.myUser.set(null);
    this.isMyUser.set(false);
    this.editMeModalOpen.set(false);
    this.members = this.buildMembersModel([]);
    this.solves = this.buildSolvesModel([]);
  }

  private fetchUserPageData(userId: string | null): Observable<UserPageData> {
    return this.getUserSeed(userId).pipe(
      switchMap((seed) => this.resolveUserSeed(seed))
    );
  }

  private resolveUserSeed(seed: UserSeed): Observable<UserPageData> {
    if (!seed.user) {
      return of({
        user: null,
        team: null,
        members: [],
        solves: [],
        me: seed.me,
      });
    }

    const selectedUser = seed.user;

    return this.getUserTeam(selectedUser, seed.team, seed.currentTeam).pipe(
      switchMap((userTeam) => this.resolveUserTeamPageData(selectedUser, userTeam, seed.me))
    );
  }

  private resolveUserTeamPageData(
    selectedUser: UserInterface,
    userTeam: UserTeam,
    me: UserInterface | null
  ): Observable<UserPageData> {
    const user = this.mergeUserFromMembers(selectedUser, userTeam.members);

    if (!userTeam.team) {
      return of({
        user,
        team: null,
        members: [],
        solves: [],
        me,
      });
    }

    return this.teamApi.getTeamSolves(userTeam.team.id).pipe(
      catchError(() => of(null)),
      map((solvesResponse) => ({
        user,
        team: userTeam.team,
        members: userTeam.members,
        solves: this.getUserSolves(user, solvesResponse?.solves ?? []),
        me,
      }))
    );
  }

  private applyPageData(res: UserPageData | null): void {
    if (!res?.user) {
      this.location.back();
      return;
    }

    this.user.set(res.user);
    this.team.set(res.team);
    this.membersList.set(res.members);
    this.solvesList.set(res.solves);
    this.myUser.set(res.me);
    this.isMyUser.set(res.user.id === res.me?.id);
    this.rebuildTableModels();
  }

  openEditMeModal() {
    if (!this.isMyUser()) {
      return;
    }

    this.editMeModalOpen.set(true);
  }

  closeEditMeModal() {
    this.editMeModalOpen.set(false);
  }

  onMeUpdated(me: MeInterface) {
    this.user.set(me.user);
    this.myUser.set(me.user);
    this.team.set(me.team);
    this.loadPage();
  }

  private getUserSeed(userId: string | null): Observable<UserSeed> {
    if (!userId) {
      return this.meApi.me().pipe(
        map((res) => ({
          user: res.user ?? null,
          team: res.team ?? null,
          currentTeam: res.team ?? null,
          me: res.user ?? null,
        }))
      );
    }

    return forkJoin({
      user: this.usersApi.getUser(userId).pipe(catchError(() => of(null))),
      me: this.meApi.me().pipe(catchError(() => of(null))),
    }).pipe(
      switchMap(({user: userResponse, me}) => {
        const user = userResponse?.user ?? null;
        const currentUser = me?.user ?? null;
        const currentTeam = me?.team ?? null;
        const seed = {
          user,
          team: this.getTeamFromUser(user, currentUser, currentTeam),
          currentTeam,
          me: currentUser,
        };

        if (seed.team || !user?.team_id) {
          return of(seed);
        }

        return this.teamApi.getTeam(user.team_id).pipe(
          catchError(() => of(null)),
          map((teamResponse) => ({
            ...seed,
            team: teamResponse?.team ?? null,
          }))
        );
      })
    );
  }

  private getUserTeam(
    user: UserInterface,
    team: TeamInterface | null,
    currentTeam: TeamInterface | null
  ): Observable<UserTeam> {
    if (team) {
      return this.teamApi.getTeamMembers(currentTeam?.id === team.id ? null : team.id).pipe(
        catchError(() => of(null)),
        map((membersResponse) => ({
          team,
          members: membersResponse?.members ?? [],
        }))
      );
    }

    if (!currentTeam) {
      return of<UserTeam>({
        team: null,
        members: [],
      });
    }

    return this.teamApi.getTeamMembers().pipe(
      catchError(() => of(null)),
      map((membersResponse) => {
        const members = membersResponse?.members ?? [];

        return members.some((member) => member.id === user.id)
          ? {
            team: currentTeam,
            members,
          }
          : {
            team: null,
            members: [],
          };
      })
    );
  }

  private getTeamFromUser(
    user: UserInterface | null,
    currentUser: UserInterface | null,
    currentTeam: TeamInterface | null
  ) {
    if (!user) {
      return null;
    }

    if (user.team) {
      return user.team;
    }

    if (currentUser?.id === user.id) {
      return currentTeam;
    }

    return null;
  }

  private mergeUserFromMembers(user: UserInterface, members: UserInterface[]) {
    const member = members.find((item) => item.id === user.id);

    return member
      ? {
        ...user,
        ...member,
      }
      : user;
  }

  private getUserSolves(user: UserInterface, solves: TeamSolveInterface[]) {
    return solves.filter((solve) => solve.user_id === user.id || solve.username === user.username);
  }

  private rebuildTableModels() {
    this.members = this.buildMembersModel(this.membersList() ?? []);
    this.solves = this.buildSolvesModel(this.solvesList() ?? []);
  }

  private buildSolvesModel(solves: TeamSolveInterface[]) {
    return createSolvesModel(solves, this.transloco, this.datePipe);
  }

  private buildMembersModel(members: UserInterface[]) {
    return createMembersModel(
      members,
      this.transloco,
      this.memberCellTemplate,
      (member) => this.getUserBadges(member)
    );
  }

  getUserBadges(user: UserInterface): MemberBadge[] {
    return getUserBadges(user, this.transloco);
  }
}

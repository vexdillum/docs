import {CommonModule} from '@angular/common';
import {HttpErrorResponse} from '@angular/common/http';
import {Component, EventEmitter, Input, OnDestroy, OnInit, Output, computed, inject, signal} from '@angular/core';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  LoadingModule,
  NotificationModule,
  TagModule,
} from 'carbon-components-angular';
import {catchError, finalize, of} from 'rxjs';
import {ChallengeInstance} from '../../entities/challenge.interface';
import {ChallengesApi} from '../../utils/api/challenges';
import {getInstanceStatusView, isHttpConnectionUrl} from '../../utils/instances/instance-status';

@Component({
  selector: 'app-challenge-instance-panel',
  imports: [
    CommonModule,
    TranslocoDirective,
    ButtonModule,
    LoadingModule,
    NotificationModule,
    TagModule,
  ],
  templateUrl: './challenge-instance-panel.html',
  styleUrl: './challenge-instance-panel.scss',
})
export class ChallengeInstancePanel implements OnInit, OnDestroy {
  private readonly challengesApi = inject(ChallengesApi);
  private readonly transloco = inject(TranslocoService);
  private timer: ReturnType<typeof setInterval> | null = null;

  readonly instance = signal<ChallengeInstance | undefined>(undefined);
  readonly isCreating = signal(false);
  readonly isDeleting = signal(false);
  readonly error = signal<string | null>(null);
  readonly copied = signal(false);
  private readonly now = signal(Date.now());

  readonly statusView = computed(() => {
    const current = this.instance();
    return getInstanceStatusView(current?.status, current?.expires_at);
  });

  readonly isHttpUrl = computed(() => {
    const connectionUrl = this.instance()?.connection_url;
    return connectionUrl ? isHttpConnectionUrl(connectionUrl) : false;
  });

  @Input({ required: true }) challengeId = '';
  @Input() hasInstance = false;

  @Input() set initialInstance(value: ChallengeInstance | undefined) {
    this.instance.set(value);
    this.error.set(null);
    this.copied.set(false);
  }

  @Output() instanceChange = new EventEmitter<ChallengeInstance | undefined>();

  ngOnInit(): void {
    this.timer = setInterval(() => this.now.set(Date.now()), 30_000);
  }

  ngOnDestroy(): void {
    if (this.timer) {
      clearInterval(this.timer);
    }
  }

  createInstance(): void {
    if (!this.challengeId || this.isCreating() || this.isDeleting()) {
      return;
    }

    this.isCreating.set(true);
    this.error.set(null);
    this.copied.set(false);

    this.challengesApi.createChallengeInstance(this.challengeId)
      .pipe(
        catchError(() => {
          this.error.set(this.transloco.translate('challenges-page.modal.instance.errors.create'));
          return of(null);
        }),
        finalize(() => this.isCreating.set(false))
      )
      .subscribe((res) => {
        if (!res) {
          return;
        }

        this.instance.set(res.instance);
        this.instanceChange.emit(res.instance);
      });
  }

  deleteInstance(): void {
    if (!this.challengeId || this.isCreating() || this.isDeleting()) {
      return;
    }

    this.isDeleting.set(true);
    this.error.set(null);
    this.copied.set(false);

    this.challengesApi.deleteChallengeInstance(this.challengeId)
      .pipe(
        catchError((error: HttpErrorResponse) => {
          if (error.status === 404) {
            this.clearInstance();
            return of({});
          }

          this.error.set(this.transloco.translate('challenges-page.modal.instance.errors.delete'));
          return of(null);
        }),
        finalize(() => this.isDeleting.set(false))
      )
      .subscribe((res) => {
        if (!res) {
          return;
        }

        this.clearInstance();
      });
  }

  openConnectionUrl(): void {
    const connectionUrl = this.instance()?.connection_url;

    if (!connectionUrl || !isHttpConnectionUrl(connectionUrl)) {
      return;
    }

    window.open(connectionUrl, '_blank', 'noopener,noreferrer');
  }

  copyConnectionUrl(): void {
    const connectionUrl = this.instance()?.connection_url;

    if (!connectionUrl) {
      return;
    }

    this.writeToClipboard(connectionUrl)
      .then(() => {
        this.copied.set(true);
        setTimeout(() => this.copied.set(false), 2_000);
      })
      .catch(() => {
        this.error.set(this.transloco.translate('challenges-page.modal.instance.errors.copy'));
      });
  }

  timeRemainingLabel(expiresAt: string): string {
    this.now();

    const expiresAtMs = new Date(expiresAt).getTime();

    if (!Number.isFinite(expiresAtMs)) {
      return '';
    }

    const diffMs = expiresAtMs - Date.now();

    if (diffMs <= 0) {
      return this.transloco.translate('instances.time.expired');
    }

    const minutes = Math.ceil(diffMs / 60_000);

    if (minutes < 60) {
      return this.transloco.translate('instances.time.minutes', { count: minutes });
    }

    const hours = Math.ceil(minutes / 60);

    if (hours < 24) {
      return this.transloco.translate('instances.time.hours', { count: hours });
    }

    return this.transloco.translate('instances.time.days', { count: Math.ceil(hours / 24) });
  }

  private clearInstance(): void {
    this.instance.set(undefined);
    this.instanceChange.emit(undefined);
  }

  private async writeToClipboard(value: string): Promise<void> {
    if (navigator.clipboard?.writeText) {
      await navigator.clipboard.writeText(value);
      return;
    }

    const textarea = document.createElement('textarea');
    textarea.value = value;
    textarea.style.position = 'fixed';
    textarea.style.opacity = '0';
    document.body.appendChild(textarea);
    textarea.select();
    document.execCommand('copy');
    document.body.removeChild(textarea);
  }
}

  @if (hasInstance) {
    <section class="instance-panel" [class.instance-panel--error]="statusView().isError">
      <div class="instance-panel__header">
        <div>
          <p class="cds--type-heading-compact-01">
            {{ t('challenges-page.modal.instance.title') }}
          </p>
          <p class="cds--type-body-compact-01 text-secondary">
            {{ t('challenges-page.modal.instance.lead') }}
          </p>
        </div>

        @if (instance()) {
          <cds-tag [type]="statusView().tagType" size="sm">
            {{ t(statusView().labelKey) }}
          </cds-tag>
        }
      </div>

      @if (error()) {
        <cds-notification
          [notificationObj]="{
            type: 'error',
            title: t('challenges-page.modal.instance.errors.title'),
            message: error()!,
            showClose: false
          }">
        </cds-notification>
      }

      @if (instance(); as activeInstance) {
        <div class="connection-block">
          <span class="cds--type-label-01 text-secondary">
            {{ t('challenges-page.modal.instance.connection') }}
          </span>
          <code class="connection-value">{{ activeInstance.connection_url }}</code>
        </div>

        <div class="instance-panel__meta">
          <span>
            {{ t('challenges-page.modal.instance.expires') }}:
            {{ activeInstance.expires_at | date:'medium' }}
          </span>
          <span>
            {{ timeRemainingLabel(activeInstance.expires_at) }}
          </span>
        </div>

        @if (statusView().isError && activeInstance.error_message) {
          <p class="instance-panel__error">
            {{ activeInstance.error_message }}
          </p>
        }

        @if (statusView().isExpired) {
          <p class="cds--type-body-compact-01 text-secondary">
            {{ t('challenges-page.modal.instance.expired') }}
          </p>
        }

        <div class="instance-panel__actions">
          @if (isHttpUrl()) {
            <button
              cdsButton="primary"
              size="sm"
              type="button"
              [disabled]="isCreating() || isDeleting()"
              (click)="openConnectionUrl()">
              {{ t('challenges-page.modal.instance.open') }}
            </button>
          }

          <button
            cdsButton="secondary"
            size="sm"
            type="button"
            [disabled]="isCreating() || isDeleting()"
            (click)="copyConnectionUrl()">
            {{ copied() ? t('challenges-page.modal.instance.copied') : t('challenges-page.modal.instance.copy') }}
          </button>

          <button
            cdsButton="danger"
            size="sm"
            type="button"
            [disabled]="isCreating() || isDeleting()"
            (click)="deleteInstance()">
            {{ isDeleting() ? t('challenges-page.modal.instance.stopping') : t('challenges-page.modal.instance.stop') }}
          </button>
        </div>
      } @else {
        <p class="cds--type-body-01 text-secondary">
          {{ t('challenges-page.modal.instance.empty') }}
        </p>
        <button
          cdsButton="secondary"
          size="sm"
          type="button"
          [disabled]="isCreating() || isDeleting()"
          (click)="createInstance()">
          {{ isCreating() ? t('challenges-page.modal.instance.starting') : t('challenges-page.modal.instance.start') }}
        </button>
      }
    </section>
  }
</ng-container>

import { Component, EventEmitter, Input, Output } from '@angular/core';
import {TilesModule} from 'carbon-components-angular';
import {ChallengePreviewInterface} from '../../entities/challenge.interface';

@Component({
  selector: 'app-challenge-preview',
  imports: [
    TilesModule
  ],
  templateUrl: './challenge-preview.html',
  styleUrl: './challenge-preview.scss',
})
export class ChallengePreview {
  @Input() challenge!: ChallengePreviewInterface;
  @Output() openChallenge = new EventEmitter<ChallengePreviewInterface>();

  onOpen() {
    this.openChallenge.emit(this.challenge);
  }
}

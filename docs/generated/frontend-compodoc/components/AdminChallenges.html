import {AfterViewInit, Component, inject, signal, TemplateRef, ViewChild} from '@angular/core';
import {FormControl, FormGroup, ReactiveFormsModule, Validators} from '@angular/forms';
import {CommonModule} from '@angular/common';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  CheckboxModule,
  ContentSwitcherModule,
  DialogModule,
  InputModule,
  LoadingModule,
  ModalModule,
  SelectModule,
  SliderModule,
  TableHeaderItem,
  TableItem,
  TableModel,
  TableModule,
} from 'carbon-components-angular';
import {ContentSwitcherOption} from 'carbon-components-angular/content-switcher';
import {AdminApi} from '../../utils/api/admin-api';
import {AdminChallenge} from '../../entities/admin/admin.interface';
import {matchesSearch} from '../../utils/search/table-search';

@Component({
  selector: 'app-admin-challenges',
  imports: [
    CommonModule,
    TranslocoDirective,
    TableModule,
    LoadingModule,
    ButtonModule,
    DialogModule,
    ModalModule,
    InputModule,
    CheckboxModule,
    SelectModule,
    ReactiveFormsModule,
    ContentSwitcherModule,
    SliderModule,
  ],
  templateUrl: './admin-challenges.html',
  styleUrl: './admin-challenges.scss',
})
export class AdminChallenges implements AfterViewInit {
  @ViewChild('linkTemplate')
  protected linkTemplate!: TemplateRef<{ data: AdminChallenge }>;

  private readonly adminApi = inject(AdminApi);
  private readonly transloco = inject(TranslocoService);

  readonly loading = signal(true);
  private allChallenges: AdminChallenge[] = [];
  private searchQuery = '';
  challenges = this.createModel([]);
  categories = signal<string[]>([]);

  selectedChallenge = signal<AdminChallenge | null>(null);
  selectedEditChallenge = signal<AdminChallenge | null>(null);
  openDeleteModal = signal(false);
  openCreateModal = signal(false);
  isEditMode = signal(false);

  readonly slugPattern = /^[a-z0-9-]+$/;

  createForm = new FormGroup({
    title: new FormControl('', [Validators.required]),
    category: new FormControl('', [Validators.required]),
    newCategory: new FormControl(''),
    description: new FormControl('', [Validators.required]),
    type: new FormControl<'static' | 'dynamic'>('static', [Validators.required]),
    flag: new FormControl('', [Validators.required]),
    image: new FormControl(''),
    points: new FormControl<[number, number]>([10, 100], [Validators.required]),
    flag_type: new FormControl('static'),
    scoring_mode: new FormControl('dynamic'),
    is_hidden: new FormControl(false),
    public: new FormControl(false),
    healthcheck_enabled: new FormControl(false),
  });

  isNewCategory = signal(false);

  get typeIndex(): number {
    return this.createForm.value.type === 'dynamic' ? 1 : 0;
  }

  isDynamicType(): boolean {
    return this.createForm.value.type === 'dynamic';
  }

  get flagTypeIndex(): number {
    return this.createForm.value.flag_type === 'regex' ? 1 : 0;
  }

  get scoringModeIndex(): number {
    return this.createForm.value.scoring_mode === 'static' ? 1 : 0;
  }

  ngAfterViewInit() {
    this.createForm.get('category')?.valueChanges.subscribe((category) => {
      this.updateNewCategoryState(category);
    });
    this.updateNewCategoryState(this.createForm.value.category);
    this.createForm.get('type')?.valueChanges.subscribe((type) => {
      this.updateTypeSpecificValidators(type ?? 'static');
    });
    this.updateTypeSpecificValidators(this.createForm.value.type ?? 'static');
    this.loadData();
  }

  private generateSlug(title: string): string {
    return title
      .toLowerCase()
      .trim()
      .replace(/[^a-z0-9]+/g, '-')
      .replace(/^-|-$/g, '');
  }

  private loadData() {
    this.loading.set(true);
    this.adminApi.getChallenges().subscribe((res) => {
      this.allChallenges = res.challenges;
      const uniqueCategories = Array.from(new Set(this.allChallenges.map(c => c.category).filter(Boolean)));
      this.categories.set(uniqueCategories);
      this.applySearch();
      this.loading.set(false);
    });
  }

  private createModel(items: AdminChallenge[]) {
    const model = new TableModel();
    model.header = [
      new TableHeaderItem({ data: this.transloco.translate('admin.challenges.columns.title') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.challenges.columns.category') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.challenges.columns.points') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.challenges.columns.status') }),
      new TableHeaderItem({ data: this.transloco.translate('admin.challenges.columns.type') }),
    ];
    model.data = items.map((c) => [
      new TableItem({
        data: c,
        template: this.linkTemplate,
      }),
      new TableItem({ data: c.category }),
      new TableItem({ data: c.current_points }),
      new TableItem({ data: c.status }),
      new TableItem({ data: c.type }),
    ]);
    return model;
  }

  isInvalid(controlName: string) {
    const control = this.createForm.get(controlName);
    return control ? control.invalid && control.touched : false;
  }

  getErrorMessage(controlName: string) {
    const control = this.createForm.get(controlName);
    if (!control?.errors || !control.touched) return '';
    if (control.errors['required']) {
      return this.transloco.translate('admin.challenges.validation.required');
    }
    if (control.errors['pattern']) {
      return this.transloco.translate('admin.challenges.validation.name.pattern');
    }
    if (control.errors['min']) {
      return this.transloco.translate('admin.challenges.validation.min');
    }
    return '';
  }

  openCreate() {
    this.isEditMode.set(false);
    this.selectedEditChallenge.set(null);
    this.createForm.reset({
      title: '',
      category: '',
      newCategory: '',
      description: '',
      type: 'static',
      flag: '',
      image: '',
      points: [10, 100],
      flag_type: 'static',
      scoring_mode: 'dynamic',
      is_hidden: false,
      public: false,
      healthcheck_enabled: false,
    });
    this.updateNewCategoryState('');
    this.updateTypeSpecificValidators('static');
    this.openCreateModal.set(true);
  }

  openEdit(challenge: AdminChallenge) {
    this.isEditMode.set(true);
    this.selectedEditChallenge.set(challenge);
    this.createForm.reset({
      title: challenge.title,
      category: challenge.category,
      newCategory: '',
      description: challenge.description ?? '',
      type: (challenge.type as 'static' | 'dynamic') ?? 'static',
      flag: challenge.flag ?? '',
      image: challenge.image ?? '',
      points: [challenge.minimum_points ?? 10, challenge.initial_points ?? 100],
      flag_type: challenge.flag_type ?? 'static',
      scoring_mode: challenge.scoring_mode ?? 'dynamic',
      is_hidden: challenge.is_hidden ?? false,
      public: challenge.public ?? false,
      healthcheck_enabled: challenge.healthcheck_enabled ?? false,
    });
    this.updateNewCategoryState(challenge.category);
    this.updateTypeSpecificValidators((challenge.type as 'static' | 'dynamic') ?? 'static');
    this.openCreateModal.set(true);
  }

  submit() {
    if (this.createForm.invalid) {
      this.createForm.markAllAsTouched();
      return;
    }
    const raw = this.createForm.value;
    const category = raw.category === '__new__' ? raw.newCategory : raw.category;
    const type = raw.type ?? 'static';

    if (raw.category === '__new__' && !category?.trim()) {
      this.createForm.get('newCategory')?.setErrors({ required: true });
      this.createForm.get('newCategory')?.markAsTouched();
      return;
    }

    if (this.isEditMode()) {
      const challenge = this.selectedEditChallenge();
      const id = challenge?.id ?? challenge?.name;
      if (!id) return;
      this.adminApi.updateChallenge(id, {
        title: this.toOptionalText(raw.title),
        category: this.toOptionalText(category),
        description: this.toOptionalText(raw.description),
        type,
        flag: type === 'static' ? this.toOptionalText(raw.flag) : undefined,
        image: type === 'dynamic' ? this.toOptionalText(raw.image) : undefined,
        initial_points: raw.points?.[1] !== undefined ? Number(raw.points[1]) : undefined,
        minimum_points: raw.points?.[0] !== undefined ? Number(raw.points[0]) : undefined,
        flag_type: type === 'static' ? raw.flag_type ?? undefined : undefined,
        scoring_mode: raw.scoring_mode ?? undefined,
        is_hidden: raw.is_hidden ?? undefined,
        public: raw.public ?? undefined,
        healthcheck_enabled: raw.healthcheck_enabled ?? undefined,
      }).subscribe(() => {
        this.openCreateModal.set(false);
        this.loadData();
      });
    } else {
      const name = this.generateSlug(raw.title ?? '');
      this.adminApi.createChallenge({
        name,
        title: raw.title!,
        category: category ?? '',
        initial_points: Number(raw.points?.[1]),
        minimum_points: Number(raw.points?.[0]),
        flag: type === 'static' ? raw.flag! : '',
        description: this.toOptionalText(raw.description),
        type,
        image: type === 'dynamic' ? this.toOptionalText(raw.image) : undefined,
        flag_type: type === 'static' ? raw.flag_type ?? undefined : undefined,
        scoring_mode: raw.scoring_mode ?? undefined,
        is_hidden: raw.is_hidden ?? false,
        public: raw.public ?? false,
        healthcheck_enabled: raw.healthcheck_enabled ?? false,
      }).subscribe(() => {
        this.openCreateModal.set(false);
        this.loadData();
      });
    }
  }

  refresh() {
    this.loadData();
  }

  onSearch(query: string) {
    this.searchQuery = query;
    this.applySearch();
  }

  deploy(challenge: AdminChallenge) {
    const id = challenge.id ?? challenge.name;
    this.adminApi.deployChallenge(id).subscribe(() => this.loadData());
  }

  stop(challenge: AdminChallenge) {
    const id = challenge.id ?? challenge.name;
    this.adminApi.stopChallenge(id).subscribe(() => this.loadData());
  }

  askDelete(challenge: AdminChallenge) {
    this.selectedChallenge.set(challenge);
    this.openDeleteModal.set(true);
  }

  askDeleteFromEdit() {
    const challenge = this.selectedEditChallenge();
    if (!challenge) return;
    this.openCreateModal.set(false);
    this.askDelete(challenge);
  }

  confirmDelete() {
    const challenge = this.selectedChallenge();
    const id = challenge?.id ?? challenge?.name;
    if (!id) return;
    this.adminApi.deleteChallenge(id, { confirm: 'DELETE_CHALLENGE' }).subscribe(() => {
      this.openDeleteModal.set(false);
      this.selectedChallenge.set(null);
      this.loadData();
    });
  }

  onTypeSwitch(option: ContentSwitcherOption) {
    const type = option.name as 'static' | 'dynamic';
    this.createForm.get('type')?.setValue(type);
    this.updateTypeSpecificValidators(type);
  }

  onFlagTypeSwitch(option: ContentSwitcherOption) {
    this.createForm.get('flag_type')?.setValue(option.name);
  }

  onScoringModeSwitch(option: ContentSwitcherOption) {
    this.createForm.get('scoring_mode')?.setValue(option.name);
  }

  private toOptionalText(value: string | null | undefined): string | undefined {
    const trimmed = value?.trim();
    if (!trimmed) {
      return undefined;
    }
    return trimmed;
  }

  private applySearch() {
    const filtered = this.allChallenges.filter((challenge) => matchesSearch([
      challenge.id,
      challenge.name,
      challenge.title,
      challenge.category,
      challenge.description,
      challenge.type,
      challenge.status,
      challenge.flag,
      challenge.flag_type,
      challenge.image,
      challenge.namespace,
      challenge.author,
      challenge.scoring_mode,
      challenge.current_points,
      challenge.initial_points,
      challenge.minimum_points,
      challenge.static_file_url,
      challenge.url,
    ], this.searchQuery));

    this.challenges = this.createModel(filtered);
  }

  private updateTypeSpecificValidators(type: 'static' | 'dynamic') {
    const flagControl = this.createForm.get('flag');
    const imageControl = this.createForm.get('image');

    if (type === 'dynamic') {
      flagControl?.clearValidators();
      imageControl?.addValidators(Validators.required);
    } else {
      flagControl?.addValidators(Validators.required);
      imageControl?.clearValidators();
    }

    flagControl?.updateValueAndValidity({ emitEvent: false });
    imageControl?.updateValueAndValidity({ emitEvent: false });
  }

  private updateNewCategoryState(category: string | null | undefined) {
    const isNewCategory = category === '__new__';
    const newCategoryControl = this.createForm.get('newCategory');

    this.isNewCategory.set(isNewCategory);

    if (isNewCategory) {
      newCategoryControl?.addValidators(Validators.required);
    } else {
      newCategoryControl?.clearValidators();
      newCategoryControl?.setValue('', { emitEvent: false });
    }

    newCategoryControl?.updateValueAndValidity({ emitEvent: false });
  }
}

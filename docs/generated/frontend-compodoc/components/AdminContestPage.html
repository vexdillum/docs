import {Component, computed, inject, signal} from '@angular/core';
import {CommonModule} from '@angular/common';
import {TranslocoDirective} from '@ngneat/transloco';
import {
  ButtonModule,
  CheckboxModule,
  InputModule,
  LoadingModule,
  SelectModule,
  TilesModule,
} from 'carbon-components-angular';
import {FormControl, FormGroup, ReactiveFormsModule} from '@angular/forms';
import {finalize} from 'rxjs';
import {AdminApi} from '../../utils/api/admin-api';
import {AdminContest, AdminContestUpdateRequest} from '../../entities/admin/admin.interface';

@Component({
  selector: 'app-admin-contest',
  imports: [
    CommonModule,
    TranslocoDirective,
    LoadingModule,
    ButtonModule,
    TilesModule,
    ReactiveFormsModule,
    InputModule,
    CheckboxModule,
    SelectModule,
  ],
  templateUrl: './admin-contest.html',
  styleUrl: './admin-contest.scss',
})
export class AdminContestPage {
  private readonly adminApi = inject(AdminApi);

  readonly loading = signal(true);
  readonly saving = signal(false);
  contest = signal<AdminContest | null>(null);

  isRunning = computed(() => this.contest()?.status === 'running');
  isFrozen = computed(() => !!this.contest()?.freeze_at);

  contestForm = new FormGroup({
    status: new FormControl<string>(''),
    starts_at: new FormControl<string>(''),
    ends_at: new FormControl<string>(''),
    freeze_at: new FormControl<string>(''),
    registration_open: new FormControl<boolean>(false),
    scoreboard_public: new FormControl<boolean>(false),
  });

  constructor() {
    this.loadData();
  }

  private loadData(): void {
    this.loading.set(true);
    this.adminApi
      .getContest()
      .pipe(finalize(() => this.loading.set(false)))
      .subscribe((res) => {
        this.contest.set(res.contest);
        this.contestForm.patchValue({
          status: res.contest.status ?? '',
          starts_at: this.toLocalDateTime(res.contest.starts_at),
          ends_at: this.toLocalDateTime(res.contest.ends_at),
          freeze_at: this.toLocalDateTime(res.contest.freeze_at),
          registration_open: res.contest.registration_open ?? false,
          scoreboard_public: res.contest.scoreboard_public ?? false,
        });
      });
  }

  private toLocalDateTime(isoString: string | undefined): string {
    if (!isoString) return '';
    const d = new Date(isoString);
    d.setMinutes(d.getMinutes() - d.getTimezoneOffset());
    return d.toISOString().slice(0, 16);
  }

  private toISOString(localDateTime: string | undefined | null): string | undefined {
    if (!localDateTime) return undefined;
    return new Date(localDateTime).toISOString();
  }

  refresh(): void {
    this.loadData();
  }

  save(): void {
    this.saving.set(true);
    this.adminApi
      .updateContest(this.buildUpdatePayload())
      .pipe(finalize(() => this.saving.set(false)))
      .subscribe(() => this.loadData());
  }

  start(): void {
    this.adminApi.startContest().subscribe(() => this.loadData());
  }

  end(): void {
    this.adminApi.endContest({ confirm: 'END_CONTEST' }).subscribe(() => this.loadData());
  }

  freeze(): void {
    this.adminApi.freezeContest().subscribe(() => this.loadData());
  }

  unfreeze(): void {
    this.adminApi.unfreezeContest().subscribe(() => this.loadData());
  }

  private buildUpdatePayload(): AdminContestUpdateRequest {
    const raw = this.contestForm.getRawValue();
    const payload: AdminContestUpdateRequest = {};
    const status = raw.status?.trim() ?? '';

    if (status) payload.status = status;
    if (raw.starts_at) payload.starts_at = this.toISOString(raw.starts_at);
    if (raw.ends_at) payload.ends_at = this.toISOString(raw.ends_at);
    if (raw.freeze_at) payload.freeze_at = this.toISOString(raw.freeze_at);
    if (raw.registration_open !== null) payload.registration_open = raw.registration_open;
    if (raw.scoreboard_public !== null) payload.scoreboard_public = raw.scoreboard_public;

    return payload;
  }
}

import {Component, inject, signal} from '@angular/core';
import {HttpErrorResponse} from '@angular/common/http';
import {ChallengePreview} from '../../widgets/challenge-preview/challenge-preview';
import {ChallengeModal} from '../../widgets/challenge-modal/challenge-modal';
import {ChallengesApi} from '../../utils/api/challenges';
import {KeyValuePipe} from '@angular/common';
import {TranslocoDirective} from '@ngneat/transloco';
import {catchError, of} from 'rxjs';
import {ChallengePreviewInterface} from '../../entities/challenge.interface';

@Component({
  selector: 'app-challenges',
  imports: [
    ChallengePreview,
    ChallengeModal,
    KeyValuePipe,
    TranslocoDirective
  ],
  templateUrl: './challenges.html',
  styleUrl: './challenges.scss',
})
export class Challenges {
   challengeApi = inject(ChallengesApi);
   challenges = signal<Record<string, ChallengePreviewInterface[]>>({});
   selectedChallenge = signal<ChallengePreviewInterface | null>(null);
   isModalOpen = signal(false);
   error = signal<'ended' | 'empty' | null>(null);

  constructor() {
     this.challengeApi.getChallenges().pipe(
       catchError((err: HttpErrorResponse) => {
         const body = err.error as { error?: unknown } | null | undefined;
         const msg = typeof body?.error === 'string' ? body.error : err.message;
         if (msg === 'contest has ended') {
           this.error.set('ended');
         }
         return of({ challenges: [] });
       })
     ).subscribe((res) => {
       if (res.challenges.length === 0 && !this.error()) {
         this.error.set('empty');
       }

       const grouped = res.challenges.reduce<Record<string, ChallengePreviewInterface[]>>(
         (acc, item) => {
           const key = item.category;

           if (!acc[key]) {
             acc[key] = [];
           }

           acc[key].push(item);
            return acc;
           },
         {}
       );

       this.challenges.set(grouped);
     });
  }

  openChallenge(challenge: ChallengePreviewInterface) {
    this.selectedChallenge.set(challenge);
    this.isModalOpen.set(true);
  }

  closeChallengeModal() {
    this.isModalOpen.set(false);
    this.selectedChallenge.set(null);
  }

  updateChallenge(updatedChallenge: ChallengePreviewInterface) {
    this.selectedChallenge.set(updatedChallenge);
    this.challenges.update((groups) => {
      const nextGroups = { ...groups };
      const categoryChallenges = nextGroups[updatedChallenge.category];

      if (!categoryChallenges) {
        return groups;
      }

      nextGroups[updatedChallenge.category] = categoryChallenges.map((challenge) =>
        challenge.id === updatedChallenge.id ? { ...challenge, ...updatedChallenge } : challenge
      );

      return nextGroups;
    });
  }
}

import {FormsModule, ReactiveFormsModule} from '@angular/forms';
import {HttpErrorResponse} from '@angular/common/http';
import {Component, EventEmitter, Input, Output, inject, signal} from '@angular/core';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {ButtonModule, InputModule, ModalModule, NotificationModule} from 'carbon-components-angular';
import {ChallengesApi} from '../../utils/api/challenges';
import {
  ChallengeInterface,
  ChallengeInstance,
  ChallengeSubmitResponseInterface,
  ChallengePreviewInterface
} from '../../entities/challenge.interface';
import {catchError, finalize, of} from 'rxjs';
import {ChallengeInstancePanel} from '../challenge-instance-panel/challenge-instance-panel';

@Component({
  selector: 'app-challenge-modal',
  imports: [
    TranslocoDirective,
    ModalModule,
    ButtonModule,
    InputModule,
    NotificationModule,
    FormsModule,
    ReactiveFormsModule,
    ChallengeInstancePanel
  ],
  templateUrl: './challenge-modal.html',
  styleUrl: './challenge-modal.scss',
})
export class ChallengeModal {
  private readonly challengesApi = inject(ChallengesApi);
  private readonly transloco = inject(TranslocoService);

  challengePreview = signal<ChallengePreviewInterface | null>(null);
  challengeDetails = signal<ChallengeInterface | null>(null)

  readonly isOpen = signal<boolean>(false);
  private challengeId: string | null = null;
  isSolved = signal<boolean>(false)

  @Input() set challenge(value: ChallengePreviewInterface | null) {
    this.challengePreview.set(value);
    this.notification.set(null);
    this.flag.set('');

    if (value && this.isOpen()) {
      this.isSolved.set(value.is_solved);
      this.loadChallenge(value.id);
      return;
    }

    if (!value) {
      this.challengeDetails.set(null);
    }
  }

  @Input() set open(value: boolean) {
    this.isOpen.set(value);

    if (value) {
      this.notification.set(null);
      this.flag.set('');
    }

    const activeChallenge = this.challengePreview();

    if (value && activeChallenge) {
      this.loadChallenge(activeChallenge.id);
    }
  }

  @Output() isCloseModal = new EventEmitter<void>();
  @Output() challengeUpdated = new EventEmitter<ChallengePreviewInterface>();

  readonly flag = signal('');
  readonly loading = signal(false);
  readonly submitting = signal(false);
  readonly notification = signal<{ type: 'error' | 'success' | 'warning'; title: string; message: string } | null>(null);

  closeModal() {
    this.isCloseModal.emit();
  }

  submitFlag() {
    const activeChallenge = this.challengeDetails();
    const flag = this.flag().trim();

    if (!activeChallenge || !flag || this.submitting()) {
      return;
    }

    this.submitting.set(true);
    this.notification.set(null);

    this.challengesApi.submitChallengeFlag(activeChallenge.id, flag)
      .pipe(
        catchError((error: HttpErrorResponse) => {
          this.handleSubmitError(error);

          return of(null);
        }),
        finalize(() => this.submitting.set(false))
      )
      .subscribe((res) => {
        if (!res) {
          return;
        }

        if (res.status === "correct") {
          this.setNotification('success', 'submission-success');
          this.loadChallenge(activeChallenge.id);
        }

        this.handleSubmitResponse(res);
      });
  }

  onInstanceChanged(instance: ChallengeInstance | undefined) {
    const activeChallenge = this.challengeDetails();

    if (!activeChallenge) {
      return;
    }

    const updatedChallenge: ChallengeInterface = {
      ...activeChallenge,
      instance,
    };

    this.challengeDetails.set(updatedChallenge);
    this.challengeUpdated.emit(updatedChallenge);
  }

  private handleSubmitError(error: HttpErrorResponse): void {
    if (error.status === 400) {
      this.setNotification('warning', 'submission-rejected');
      return;
    }

    if (error.status === 403) {
      this.setNotification('warning', 'team-page-required');
      return;
    }

    this.setNotification('error', 'submission-failed');
  }

  private loadChallenge(challengeId: string) {
    this.challengeId = challengeId;
    this.loading.set(true);

    this.challengesApi.getChallengeById(challengeId)
      .pipe(
        catchError(() => {
          this.setNotification('error', 'challenge-unavailable');

          this.challengeDetails.set(null);
          return of(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => {
        if (this.challengeId !== res?.challenge.id) {
          return;
        }

        this.challengeDetails.set(res.challenge);
        this.isSolved.set(res.challenge.is_solved);
      });
  }

  private handleSubmitResponse(response: ChallengeSubmitResponseInterface) {
    const current = this.challengeDetails();

    if (!current) {
      return;
    }

    if (response.status === 'correct' || response.status === 'already_solved') {
      const updatedChallenge: ChallengeInterface = {
        ...current,
        is_solved: true
      };

      this.challengeDetails.set(updatedChallenge);
      this.challengeUpdated.emit(updatedChallenge);
      this.setSuccessfulSubmitNotification(response);
      return;
    }

    this.setNotification('error', 'wrong-flag');
  }

  private setSuccessfulSubmitNotification(response: ChallengeSubmitResponseInterface): void {
    if (response.status === 'correct') {
      this.setNotification('success', 'correct-flag', {
        points: response.points_awarded,
        score: response.team_score
      });
      return;
    }

    this.setNotification('success', 'already-solved');
  }

  private setNotification(
    type: 'error' | 'success' | 'warning',
    key: string,
    params?: Record<string, unknown>
  ): void {
    this.notification.set({
      type,
      title: this.translateNotification(`${key}.title`, params),
      message: this.translateNotification(`${key}.message`, params),
    });
  }

  private translateNotification(key: string, params?: Record<string, unknown>) {
    return this.transloco.translate(`challenges-page.modal.notifications.${key}`, params);
  }
}

        <h3 cdsModalHeaderHeading>{{ challengeDetails()?.title || challengePreview()?.title }}</h3>
      </cds-modal-header>
      <section cdsModalContent [hasForm]="true">
        <div class="w-full flex flex-col gap-[12px]">
        @if (notification()) {
          <cds-notification
            [notificationObj]="{
              type: notification()!.type,
              title: notification()!.title,
              message: notification()!.message,
              showClose: false
            }">
          </cds-notification>
        } @else if (isSolved()) {
          <cds-notification
            [notificationObj]="{
              type: 'success',
              title: t('challenges-page.modal.notifications.submission-success.title'),
              message: t('challenges-page.modal.notifications.submission-success.message'),
              showClose: false
            }">
          </cds-notification>
        }

        @if (loading()) {
          <h1 cdsModalContentText class="cds--modal-content__regular-content modal-text">
            {{ t('challenges-page.modal.loading') }}
          </h1>
        } @else if (challengeDetails(); as activeChallenge) {
          <h1 cdsModalContentText class="cds--modal-content__regular-content modal-text">
            {{ challengeDetails()?.description }}
          </h1>

          <app-challenge-instance-panel
            [challengeId]="activeChallenge.id"
            [hasInstance]="activeChallenge.has_instance"
            [initialInstance]="activeChallenge.instance"
            (instanceChange)="onInstanceChanged($event)">
          </app-challenge-instance-panel>

          @if (!isSolved()) {
            <cds-text-label>
              <span class="cds--type-body-01">
                flag
              </span>
              <input
                type="text"
                cdsText
                size="md"
                [placeholder]="t('challenges-page.modal.flag-placeholder')"
                [ngModel]="flag()"
                (ngModelChange)="flag.set($event)">
            </cds-text-label>
          }
        }
        </div>
      </section>
      <cds-modal-footer>
        <button class="cds--btn cds--btn--secondary" (click)="closeModal()">
          {{ t('challenges-page.modal.close') }}
        </button>
        @if (!isSolved()) {
          <button
            class="cds--btn cds--btn--primary"
            modal-primary-focus
            [disabled]="loading() || submitting() || !flag().trim()"
            (click)="submitFlag()">
            {{ t('challenges-page.modal.submit') }}
          </button>
        }
      </cds-modal-footer>
    </cds-modal>
  }
</ng-container>

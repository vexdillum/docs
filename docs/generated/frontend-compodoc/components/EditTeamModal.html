import {Component, EventEmitter, Input, Output, inject, signal} from '@angular/core';
import {HttpErrorResponse} from '@angular/common/http';
import {FormControl, FormGroup, ReactiveFormsModule, Validators} from '@angular/forms';
import {TranslocoDirective, TranslocoService} from '@ngneat/transloco';
import {
  ButtonModule,
  InputModule,
  ModalModule,
  NotificationModule
} from 'carbon-components-angular';
import {catchError, finalize, of} from 'rxjs';
import {TeamInterface, UpdateTeamInterface} from '../../entities/team.interface';
import {TeamApi} from '../../utils/team/team-api';
import {controlInvalid, getControlErrorKey} from '../../utils/auth/auth-form-validators';

type EditTeamControlName = 'name' | 'password' | 'category' | 'affiliation';

const TEAM_NAME_MAX_LENGTH = 64;
const TEAM_PASSWORD_MIN_LENGTH = 8;
const TEAM_AFFILIATION_MAX_LENGTH = 128;

@Component({
  selector: 'app-edit-team-modal',
  imports: [
    ButtonModule,
    InputModule,
    ModalModule,
    NotificationModule,
    ReactiveFormsModule,
    TranslocoDirective
  ],
  templateUrl: './edit-team-modal.html',
  styleUrl: './edit-team-modal.scss',
})
export class EditTeamModal {
  private readonly teamApi = inject(TeamApi);
  private readonly transloco = inject(TranslocoService);

  readonly isOpen = signal<boolean>(false);
  readonly team = signal<TeamInterface | null>(null);
  readonly canEdit = signal<boolean>(false);
  readonly loading = signal<boolean>(false);
  readonly deleting = signal<boolean>(false);
  readonly notification = signal<{ type: 'error' | 'success'; title: string; message: string } | null>(null);

  readonly form = new FormGroup({
    name: new FormControl('', [
      Validators.required,
      Validators.maxLength(TEAM_NAME_MAX_LENGTH),
    ]),
    password: new FormControl('', [
      Validators.minLength(TEAM_PASSWORD_MIN_LENGTH),
    ]),
    category: new FormControl(''),
    affiliation: new FormControl('', [
      Validators.maxLength(TEAM_AFFILIATION_MAX_LENGTH),
    ]),
  });

  @Input() set open(value: boolean) {
    this.isOpen.set(value);

    if (value) {
      this.notification.set(null);
      this.patchForm();
    }
  }

  @Input() set activeTeam(value: TeamInterface | null) {
    this.team.set(value);

    if (this.isOpen()) {
      this.patchForm();
    }
  }

  @Input() set captain(value: boolean) {
    this.canEdit.set(value);
  }

  @Output() isCloseModal = new EventEmitter<void>();
  @Output() teamUpdated = new EventEmitter<TeamInterface>();
  @Output() teamDeleted = new EventEmitter<void>();

  closeModal() {
    this.isCloseModal.emit();
  }

  isInvalid(controlName: EditTeamControlName) {
    return controlInvalid(this.form.get(controlName));
  }

  getErrorMessage(controlName: EditTeamControlName) {
    const errorKey = getControlErrorKey(this.form.get(controlName));

    return errorKey ? this.transloco.translate(`edit-team-modal.validation.${controlName}.${errorKey}`) : '';
  }

  saveTeam() {
    if (!this.canEdit() || this.form.invalid || this.loading()) {
      this.form.markAllAsTouched();
      return;
    }

    const activeTeam = this.team();

    if (!activeTeam) {
      return;
    }

    const payload = this.getPayload(activeTeam);

    if (!payload) {
      this.closeModal();
      return;
    }

    this.loading.set(true);
    this.notification.set(null);

    this.teamApi.updateTeam(payload)
      .pipe(
        catchError((error: HttpErrorResponse) => {
          this.showRequestError(error.status, 'update');
          return of(null);
        }),
        finalize(() => this.loading.set(false))
      )
      .subscribe((res) => {
        if (!res?.team) {
          return;
        }

        this.teamUpdated.emit(res.team);
        this.closeModal();
      });
  }

  deleteTeam() {
    if (!this.canEdit() || this.deleting()) {
      return;
    }

    this.deleting.set(true);
    this.notification.set(null);

    this.teamApi.deleteTeam()
      .pipe(
        catchError((error: HttpErrorResponse) => {
          this.showRequestError(error.status, 'delete');
          return of(null);
        }),
        finalize(() => this.deleting.set(false))
      )
      .subscribe((res) => {
        if (!res) {
          return;
        }

        this.teamDeleted.emit();
        this.closeModal();
      });
  }

  private patchForm() {
    const activeTeam = this.team();

    this.form.reset({
      name: activeTeam?.name ?? '',
      password: '',
      category: activeTeam?.category ?? '',
      affiliation: activeTeam?.affiliation ?? '',
    });
  }

  private getPayload(activeTeam: TeamInterface): UpdateTeamInterface | null {
    const formValue = this.form.getRawValue();
    const name = formValue.name?.trim();

    if (!name) {
      return null;
    }

    const payload: UpdateTeamInterface = { name };
    let hasChanges = name !== activeTeam.name;

    const category = formValue.category?.trim();
    const affiliation = formValue.affiliation?.trim();
    const password = formValue.password?.trim();

    if (password) {
      payload.password = password;
      hasChanges = true;
    }

    if (category !== activeTeam.category) {
      payload.category = category;
      hasChanges = true;
    }

    if (affiliation !== activeTeam.affiliation) {
      payload.affiliation = affiliation;
      hasChanges = true;
    }

    return hasChanges ? payload : null;
  }

  private showRequestError(status: number, action: 'update' | 'delete') {
    const messageKey = status === 0
      ? 'no-connection'
      : status === 400 || status === 401 || status === 403 || status === 404 || status === 409
        ? `${action}-invalid`
        : 'universal';

    this.notification.set({
      type: 'error',
      title: this.transloco.translate('edit-team-modal.notifications.error.title'),
      message: this.transloco.translate(`edit-team-modal.notifications.error.${messageKey}`),
    });
  }

}
